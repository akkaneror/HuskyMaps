566        /** Checks whether the address selection fields have been changed to something different
567         * than the most recent FindDirections call */
568        private boolean checkAddrInputFields(){
569            boolean same = textFieldCurrents.get(jTextFieldStart).equals(jTextFieldStart.getText()) &&
570                                    textFieldCurrents.get(jTextFieldEnd).equals(jTextFieldEnd.getText()) &&
571                                    jComboBoxFormat.getSelectedIndex() == selectedFormat;
572            
573            if(DEBUG){
574                    outputResults("lasts:");
575                    outputResults(textFieldCurrents.get(jTextFieldStart));
576                            outputResults(textFieldCurrents.get(jTextFieldEnd));
577                            outputResults(""+jComboBoxFormat.getSelectedIndex()); 
578                            outputResults("nows:");
579                            outputResults(jTextFieldEnd.getText());
580                            outputResults(jTextFieldStart.getText());
581                            outputResults(""+selectedFormat);
582            }
583            
584            if(!same){
585                    jButtonGetDirections.setText("Find Directions!");
586            }else{
587                    jButtonGetDirections.setText("Refresh Map");
588            }
589            return same;
590        }

611        /** Interprets the database-loading text fields to read in a database, including 
612         * instantiating a DirectionsFinder object*/
613        private void loadDatabase(){
614            
615            String db = jFieldDBFile.getText().trim();
616            boolean badzip = false;
617            
618            Collection<String> zf = new LinkedList<String>();
619            
620            // handle zipcodes if zip filtering specified
621            if(jCheckBoxZipFilter.isSelected()){
622                    
623                    String zips = jTextFieldZipcodes.getText().trim();
624                    
625                    String[] z = zips.split(",");
626                    for(int i = 0; i < z.length; i++){
627                            zf.add(z[i].trim());
628                    }
629                    
630                    for(String s : zf){
631                            if(s.length() != 5){
632                                    badzip = true;
633                            }
634                            try{
635                                    Integer.parseInt(s);
636                            }catch(NumberFormatException ex){
637                                    badzip = true;
638                            }
639                    }
640                    
641            }
642            
643            // if the zipcode input is bad
644            if(badzip){
645                    outputResults("Please check your zipcodes.\nA zipcode must be a 5 digit number...\n");
646                    jButtonDBLoad.setForeground(new java.awt.Color(255, 37, 37));
647                    jButtonDBLoad.setText("Check Zipcodes");
648            }else{
649                    // attempt to load database
650                    try{
651                            
652                            if(DEBUG_SETDB) db = DEBUG_DB;
653                            
654                            jButtonDBLoad.setForeground(Color.BLACK);
655                            jButtonDBLoad.setText("Loading...");
656                            virtualGPS = DirectionsFinder.getDirectionsFinder(db, zf);
657                        jButtonDBLoad.setText("Load Database");
658                        
659                            jPanelAddr.setVisible(true);
660                            jPanelDB.setVisible(false);
661                        jButtonDBToAddr.setVisible(true);
662                            
663                    }catch(InvalidDatabaseException ex){
664                            outputResults("Please check the database path.\nThe path name must refer " +
665                                            "to a valid Tiger Database...\n");
666                        jButtonDBLoad.setForeground(new java.awt.Color(255, 37, 37));
667                        jButtonDBLoad.setText("Check Database");
668                    }
669            }
670            
671        }

674        /** handles all directions calculations, including calling DirectionsFinder.getDirections() */
675        private void calculateDirections(){
676            
677            Address start = parseAddress(jTextFieldStart);
678            Address end = parseAddress(jTextFieldEnd);
679            if(DEBUG_SETDB){  start = DEBUG_START;   end = DEBUG_END; }
680            
681            
682            if(textFieldDefaults.get(jTextFieldStart).equals(jTextFieldStart.getText())  ||
683                            textFieldDefaults.get(jTextFieldEnd).equals(jTextFieldEnd.getText())){
684                    outputResults(addressError(null, -5));
685                    
686            }else if(start == null || end == null){
687                    outputResults(addressError(null, -4));
688                    
689            }else{
690                            try{
691                                    if(!checkAddrInputFields()){
692                                            RouteFormatter format = getTravelFormat();
693            
694                                            currDirections = virtualGPS.getDirections(start, end, format);
695                                            // prepare the map view
696                                            if(!jFrameMap.isShowing()){
697                                            jFrameMap.setVisible(true);
698                                            jFrameMap.setBounds(this.getX()+this.getWidth(), this.getY(), 
699                                                            jFrameMap.getPreferredSize().width,
700                                                            jFrameMap.getPreferredSize().height);
701                                            }
702                                            
703                                            outputResults(formatDirections(format));
704                                            
705                                            setAddrInputFields();
706                                            checkAddrInputFields();
707                                            
708                                    }else{
709                                            mapPanel.requestFocus();
710                                    }
711                                    
712                            generateMap();
713                            
714                            
715                            }catch(InvalidAddressException ex){
716                                    
717                            int error = virtualGPS.checkAddress(start);
718                            if(error != DirectionsFinder.ADDRESS_VALID){       // if start address is bad
719                                    outputResults(addressError(start, error));
720                            
721                            }else{
722                                    error = virtualGPS.checkAddress(end);
723                                    outputResults(addressError(end, error));
724                            }
725                            
726                            }catch(NoPathException ex){
727                                    
728                                    outputResults("No path found:\nFrom "+start+"\nTo "+end+"\n");
729                            }
730            }
731        
732        }

738        /** Interprets the specified text field as an address, and returns the resulting Address 
739         * @return null if the address format is invalid
740         * */
741        private Address parseAddress(javax.swing.JTextField tx){
742            
743            String in = tx.getText().trim();
744            
745            String[] arr = in.split(" ");
746            if(arr.length < 3){
747                    return null;
748            }
749            
750            int num;
751            String zip;
752            String name;
753            try{
754                    num = Integer.parseInt(arr[0]);
755                    zip = arr[arr.length - 1];
756                    name = "";
757                    for(int i = 1; i < arr.length -1; i++){
758                            if(arr[i].length() != 0){
759                                    name += " "+arr[i];
760                            }
761                    }
762      
763            }catch(Exception ex){
764                    return null;
765            }
766            
767            return new Address(num, name.substring(1), zip);
768        }

770        /** helper to handle Address error-reporting 
771         * @return a String of the error report */
772        private String addressError(Address addr, int cause){
773    
774            if(cause == DirectionsFinder.ADDRESS_VALID){
775                    return "Should not have thrown error for "+addr+"\n";
776            }else if(cause == DirectionsFinder.ADDRESS_ZIPCODE_INVALID){
777                            return "No such zipcode: "+addr+"\n";
778                    }else if(cause == DirectionsFinder.ADDRESS_STREET_INVALID){
779                            return "No such street: "+addr+"\n";
780                    }else if(cause == DirectionsFinder.ADDRESS_NUM_INVALID){
781                            return "No such number: "+addr+"\n";
782                    }else if(cause == DirectionsFinder.ADDRESS_NULL){
783                            return "Address "+addr+" was not entered in the correct format.\nEnter " +
784                                            "the address number, the street name, then a 5 " +
785                                            "digit zipcode...\n";
786                    }else{  // address is null
787                            return "Complete your address selection...\n";
788                    }
789        }

804        /** Returns the current directions instructions to output-ready form */
805        private String formatDirections(RouteFormatter formatter)
806        {
807            if(currDirections == null){
808                    throw new IllegalStateException("directions object should not be null at this point");
809            }
810            
811                    // start the string
812            StringBuilder instruct = new StringBuilder();
813                    instruct.append("Start at "+currDirections.getStart()+"\n");
814                    
815                    Iterator<String> iter = currDirections.iterator();
816                    while(iter.hasNext()){
817                            instruct.append(iter.next()+"\n");
818                    }
819                    //distinguish between miles and minutes
820                    if(formatter instanceof WalkingRouteFormatter){
821                            //output.println("Trip time: " + (Math.round(navigate.getLength()) * 20) + " minutes");
822                            
823                            instruct.append("Trip time: "+Math.round(currDirections.getLength()*20)+" minutes\n\n\n");
824                    }else if(formatter instanceof DrivingRouteFormatter){
825                            //DecimalFormat printer = new DecimalFormat("#.#"); // <<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<
826                            
827                            instruct.append("Trip length: "+(Math.round(currDirections.getLength()*10))/10.0+" miles\n\n\n");
828                    }else{
829                            throw new IllegalArgumentException("Directions implementor class not recognized");
830                    }
831                    return instruct.toString();
832        }

838        /** 
839         * Draws the graphical map centered on the current directions path (currDirections).
840         *   It gives at least a 5% buffer of space from the border of the window to any point 
841         *   of the path.  Also fills in all other streets in the view that connect somehow to 
842         *   any point of the current path.  Also maintains a minimum viewing boundary of 2 degrees
843         *   latitude and 2 degrees longitude (for mono- latitude or longitude paths).  
844         */
845        private void generateMap(){
846            if(currDirections != null){
847    
848                    Point edge = new Point(mapPanel.getWidth(), mapPanel.getHeight());
849                    mapPanel.paintImmediately(0, 0, edge.x, edge.y);
850                    
851                    
852                    // setup advanced graphics object
853                    Graphics2D g = (Graphics2D)mapPanel.getGraphics();
854                    g.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
855                    g.setRenderingHint(RenderingHints.KEY_STROKE_CONTROL, RenderingHints.VALUE_STROKE_NORMALIZE);
856    
857                    
858                    Route rt = currDirections.getRoute();
859            
860                    boolean singleStreet = rt.getGeoSegments().size() == 1;
861                    Route contextRoute = (!singleStreet ? rt : 
862                                    new Route( virtualGPS.findFullSegment(rt.getEndingGeoSegment()) ) );
863                    
864                    // calculate the longitude and latitude bounds
865                    int[] bounds = calculateCoverage( contextRoute , edge);
866                    
867                    Collection<StreetSegment> map = virtualGPS.getContext(bounds, contextRoute);
868                    
869                    
870                    // draw every segment of the map
871                    g.setColor(Color.GRAY);
872                    for(GeoSegment gs : map){
873                            drawSegment(g, bounds, edge, gs);
874                    }
875                    
876                    // draw the path over the produced map in BLUE
877                    g.setColor(Color.BLUE);
878                    for(GeoFeature gf : rt.getGeoFeatures()){
879                        for(GeoSegment gs : gf.getGeoSegments()){
880                                    drawSegment(g, bounds, edge, gs);
881                        }
882                        Point start = derivePoint(bounds, edge, gf.getStart());
883                        Point end = derivePoint(bounds, edge, gf.getEnd());
884                        
885                        int dX = (int)(Math.abs(start.x - end.x) / 2.0);
886                        int dY = (int)(Math.abs(start.y - end.y) / 2.0);
887    
888                        if(!jCheckBoxHideStreetNames.isSelected()){
889                            g.setFont(new Font(g.getFont().getFontName(), Font.BOLD, g.getFont().getSize()));
890                            g.drawString(gf.getName(), start.x + (start.x<end.x ? dX : -dX),
891                                            start.y + (start.y<end.y ? dY : -dY));
892                        }
893                    }
894            }
895        }

898        /** Draws the specified GeoSegment in the current Map View panel */
899        private void drawSegment(Graphics2D g, int[] bounds, Point edge, GeoSegment gs){
900            StreetSegment stS = ((StreetSegment)gs);
901            float size;
902            if(stS.getStreetClass().equals(StreetClassification.PRIM_HWY)){
903                    size = 4;
904            }else if(stS.getStreetClass().equals(StreetClassification.SEC_HWY)){
905                    size = 3;
906            }else if(stS.getStreetClass().equals(StreetClassification.LOCAL_ROAD)){
907                    size = 2;
908            }else{  // UKNOWN_ROAD
909                    size = 1;
910            }
911            g.setStroke(new BasicStroke(size, BasicStroke.CAP_ROUND, BasicStroke.JOIN_ROUND));
912            Point p1 = derivePoint(bounds, edge, gs.getP1());
913            Point p2 = derivePoint(bounds, edge, gs.getP2());
914            g.drawLine(p1.x, p1.y, p2.x, p2.y);
915        }

918        /** Helper to calculate the Latitude and Longitude limits of the Map View
919         *  panel based on the specified route.  The limits should be calculated
920         *  based on the current Map View window's size, and give at least a 5% buffer 
921         *  of space from the boarder to any point of the route. */
922        private int[] calculateCoverage(Route rt, Point edge){
923    
924            int[] latLonMinMax = new int[4];
925            
926            // figure out the bounding box of the route
927            latLonMinMax[0] = rt.getStart().getLatitude();
928            latLonMinMax[1] = latLonMinMax[0];
929            latLonMinMax[2] = rt.getStart().getLongitude();
930            latLonMinMax[3] = latLonMinMax[2];
931            
932            for(GeoSegment gs : rt.getGeoSegments()){
933                    int temp = gs.getP2().getLatitude();
934                    latLonMinMax[0] = latLonMinMax[0] > temp ? temp : latLonMinMax[0];
935                    latLonMinMax[1] = latLonMinMax[1] < temp ? temp : latLonMinMax[1];
936                    
937                    temp = gs.getP2().getLongitude();
938                    latLonMinMax[2] = latLonMinMax[2] > temp ? temp : latLonMinMax[2];
939                    latLonMinMax[3] = latLonMinMax[3] < temp ? temp : latLonMinMax[3];
940            }
941            
942            
943            // now use aspect ratio of the window to get bounds of the display
944            int diffLat = latLonMinMax[1] - latLonMinMax[0];
945            int diffLon = latLonMinMax[3] - latLonMinMax[2];
946            
947            // the longest aspect will be a larger ratio
948            double horzAsp = diffLon / edge.getX();
949            double vertAsp = diffLat / edge.getY();
950            
951            double expandLat = 0.0;
952            double expandLon = 0.0;
953            if(horzAsp > vertAsp){
954                    double factor = horzAsp/vertAsp;
955                    // calculate the expansion of the short dimension
956                    expandLat = ((diffLat * factor - diffLat) / 2.0);
957                    expandLat += ((diffLat * factor) / 20.0);  // 20.0 for 5%
958                    // long dimension
959                    expandLon = diffLon / 20.0;
960            }else{
961                    double factor = vertAsp/horzAsp;
962                    // calculate the expansion of the short dimension
963                    expandLon = ((diffLon * factor - diffLon) / 2);
964                    expandLon += ((diffLon * factor) / 20.0);
965                    // long dimension
966                    expandLat = diffLat / 20.0;
967            }
968            
969            latLonMinMax[0] -= ( ((int)expandLat) != 0 ? ((int)expandLat) : 2);
970            latLonMinMax[1] += ( ((int)expandLat) != 0 ? ((int)expandLat) : 2);
971            latLonMinMax[2] -= ( ((int)expandLon) != 0 ? ((int)expandLon) : 2);
972            latLonMinMax[3] += ( ((int)expandLon) != 0 ? ((int)expandLon) : 2);
973            
974            return latLonMinMax;
975        }

978        /** helper to determine the x,y coordinates of a GeoPoint */
979        private Point derivePoint(int[] minMaxLatLon, Point edge, GeoPoint p){
980            int y = (int)((minMaxLatLon[1] - p.getLatitude()) /
981                            ((double)(minMaxLatLon[1] - minMaxLatLon[0])) *
982                            (edge.getY()));
983            
984            int x = (int)((p.getLongitude() - minMaxLatLon[2]) /
985                            ((double)(minMaxLatLon[3] - minMaxLatLon[2])) *
986                            (edge.getX()));
987            
988            return new Point(x, y);
989        }
