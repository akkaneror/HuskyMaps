014    /**
015     * <p>
016     * A StreetPath is a sequence of StreetSegments that represent a continuous 
017     * navigational path across the earth's surface.  It's cost is defined
018     * as the total distance traversed by the path's StreetSegments so far, plus
019     * the geographic distance to the nearest goal segment. 
020     * </p>
021     * 
022     * <p>
023     * A StreetPath is immutable. A new StreetPath is returned through
024     * the extend path operation.
025     * </p>
026     * 
027     * <p>
028     * The elements of this path are StreetSegments.  For any single segment of the path,
029     * the segment's start point must be equal to the end point of the previous 
030     * segment, and it's end point must be equal to the start point of the next segment
031     * in the path.
032     * </p>
033     * 
034     * <p>
035     * StreetPath contains most implementation of the advanced features of HuskyMaps,
036     * controlled statically by the fields of ps7.EnvironmentVariables.  See documentation
037     * for details.
038     * </p>
039     *
040     * Specfields inherited from Path:
041     * @specfield  cost :     double       // cost of traversing this path.
042     * @specfield  elements : sequence     // sequence of nodes in this path.
043     * 
044     * 
045     * Local specfields
046     * @specfield  node : StreetSegment             // the end node of this path
047     * @specfield  goals : Set<StreetEntry>   // collection of goal StreetEntrys
048     *                                                                              // for this path instance 
049     * @specfield  weightedLength : double  // cost-accounting length of this path
050     * @specfield  hash : int                       // the hashcode of this path 
051     * @specfield  start :  StreetEntry             // StreetEntry of the starting address
052     * 
053     * @specfield  geographicMap : Graph<StreetSegment>  // Reference to the context 
054     *                                                                                              //graph to determine intersections
055     */
056    
057    
058    public class StreetPath implements Path<StreetSegment, StreetPath> {
059    
060            
061            /*
062             * Abstraction Function
063             * AF(c) = < cost, elements >  where
064             *              cost     = c.cost
065             *              elements =      [c.node]                                        if c.path == null
066             *                               ||     [c.node] + c.path.elements              if c.path != null
067             * 
068             *              Intersections:
069             *                      An element traverses an intersection if 
070             *                      geographicMap.edgeSet( element ).size() > 1
071             *              
072             * 
073             * 
074             * Representation Invariant 
075             *              (this.node != null && this.goals != null && !this.goals.isEmpty() ) &&
076             *              (this.path == null) ==> (this.cost == this.node.cost) &&
077             *              (this.path != null) ==> (this.cost == this.node.cost + this.path.cost)
078             * 
079             *              In a path of n elements, for element e, {1 < e < n-1} :
080             *                      e.getP1().equals( [e-1].getP2() ) && e.getP2().equals( [e+1].getP1() )
081             *                      that is for any element, it's start point must be equal to the end point
082             *                      of the previous element, and it's end point must be equal to the start
083             *                      point of the next element
084             */
085            
086            /** Switch to turn on checks of the representation invariant */
087            private static final boolean CHECK_REP = false;
088    
089    
090            
091            /** The absolute closest distance of any path object to its goal */
092            public static double CLOSEST_PATH_DISTANCE = Double.MAX_VALUE;
093            
094        /** The StreetSegment at the end of the path. */
095            private final StreetSegment node;
096        /** A StreetPath which, when extended with 'node' at the end,
097         *  is equal to this.  May be null iff this path has only 1 element. */
098        private final StreetPath path;
099        /** The length of this StreetPath in weighted miles as calculated taking into account 
100         * street classification and the number of intersections traversed. Weighted miles
101         * are representative of travel time, not true travel distance.*/
102        private final double weightedLength;
103        /** The cost of this StreetPath taking into account street classification and
104         * the number of intersections traversed. */
105        private final double cost;
106        /** The hashcode of this StreetPath */
107        private final int hash;
108        
109        /** The StreetEntry of the starting address */
110        private final StreetEntry start;
111        /** Reference to the context graph to determine intersections */
112            private final Graph<StreetSegment> geographicMap;
113            
114        /** Set of goal StreetSegments of this Path, to compute  */
115            private final Set<StreetEntry> goals; // for computing cost
116    
117            
118            
119            /**
120             * A new StreetPath with the specified set of StreetSegments acting as the
121             * end goals for this StreetPath instance.
122             * @requires start != null && goals != null && !goals.isEmpty() && geographicMap != null
123             * @param goals a set of StreetEntrys as the goals of this path
124             * @param the Graph to operate over
125             * @effects Creates a new single-element StreetPath with the element contained within
126             * the specified StreetEntry and no additional elements. 
127             */
128        public StreetPath(StreetEntry start, Set<StreetEntry> goals,
129                    Graph<StreetSegment> geographicMap) {
130            this(start.getStreet(), null, goals, start, geographicMap);
131                    
132                    // this path has reached a goal segment, so reset the absolute closest path distance
133                    CLOSEST_PATH_DISTANCE = Double.MAX_VALUE;
134        }
135            
136        /**
137         * Constructs a new StreetPath with the specified arguments
138             * @requires node != null && goals != null && !goals.isEmpty() && geographicMap != null
139             * @param node the end segment of this path
140             * @param path the rest of the path
141             * @param goals the set of goals for this path
142             * @param start the start Entry for this path
143             * @param the Graph to operate over
144             * @effects Creates a new StreetPath which is the specified path
145             * plus the specified node added as it's new end element.
146             */
147        @SuppressWarnings("unused")
148            private StreetPath(StreetSegment node, StreetPath path, Set<StreetEntry> goals,
149                    StreetEntry start, Graph<StreetSegment> geographicMap){
150            if (node == null || goals == null || start == null || geographicMap == null) {
151                throw new IllegalArgumentException("no specified arguments can be null");
152            }else if(goals.isEmpty()){
153                throw new IllegalArgumentException("must have at least one goal segment");  
154            }
155            this.node = node;
156            this.path = path;
157            this.goals = goals;
158            this.start = start;
159                    this.geographicMap = geographicMap;
160    
161            // get length in weighted miles of the path so far
162                    double length = (path != null) ? path.weightedLength : 0;
163                    if(EnvironmentVariables.INTERSECTION_TRAVERSAL_ON){
164                            length += (geographicMap.edgeSet(node).size() > 1 ? 
165                                            EnvironmentVariables.INTERSECTION_TRAVERSAL_COST : 0);
166                                            // add the cost of traversing an intersection if this node has 
167                                            // more than 1 edge in the graph, otherwise it's just a continuing segment.
168                    }
169                    length += (EnvironmentVariables.STREET_CLASSIFICATION_ON ? 
170                                    parseStreetClass(node) : 1) * 
171                                    ((path == null && EnvironmentVariables.FRACTIONAL_SEGMENTS_ON) ? 
172                                    start.getFracLength() : node.getLength());
173                    this.weightedLength = length;
174                    
175                    
176                    double distance = Double.MAX_VALUE;
177                    for(StreetEntry se : goals){
178                            StreetSegment gs = se.getStreet();
179                            if(node.equals(gs)){    // if this path has reached a goal segment, then 
180                                                                                                    // cost can double as the final length 
181                                    distance = 0;
182                                    break;
183                            }
184                            distance = Math.min(distance, node.getP2().distanceTo(gs.getP1()) +
185                                            (EnvironmentVariables.FRACTIONAL_SEGMENTS_ON ? se.getFracLength() : 0));
186                                            // factor in the distance to the address entering the goal segment  
187                    }
188                    
189                    this.cost = this.weightedLength + calculateDistanceCostFactor(distance);
190                                    // multiply this segment by its 
191    ;
192                    
193                    this.hash = node.hashCode() + (this.path==null ? 1 : path.hashCode());
194                    
195                    if(CHECK_REP) checkRep();
196        }
197    
198        /** helper to assign an extra factor of cost to a street segment based on its 
199         * Street Classification */
200        private double parseStreetClass(StreetSegment street){
201            StreetClassification sc = street.getStreetClass();
202            if(sc == StreetClassification.PRIM_HWY){
203                    return EnvironmentVariables.PRIM_HWY_COST;
204            }else if(sc == StreetClassification.SEC_HWY){
205                    return EnvironmentVariables.SEC_HWY_COST;
206            }else if(sc == StreetClassification.LOCAL_ROAD){
207                    return EnvironmentVariables.LOCAL_ROAD_COST;
208            }else{
209                    return EnvironmentVariables.UNKNOWN_ROAD_COST;
210            }
211        }
212        
213        /**
214         * Calculates the distance-to-goal cost factor for this StreetPath based on
215         * this path's distance to a goal and the absolute shortest distance out of
216         * all StreetPath objects.  Also relies on system environment variables.
217         * @param distance this path's shortest distance to a goal
218         * @return the distance-to-goal cost factor to be used in determining this path's cost. 
219         */
220        @SuppressWarnings("unused")
221            private double calculateDistanceCostFactor(double distance){
222            if(!EnvironmentVariables.STREET_CLASSIFICATION_ON &&
223                    !EnvironmentVariables.INTERSECTION_TRAVERSAL_ON ){
224                            return distance;
225                    }
226            
227            double distanceFactor = EnvironmentVariables.DISTANCE_TO_GOAL_COST_FACTOR;
228            
229            // determine cost optimization for this path 
230            double powerFactor = 1.0;
231            if(distance < CLOSEST_PATH_DISTANCE)
232            {
233                    CLOSEST_PATH_DISTANCE = distance;
234                    
235            // if the distance falls outside of the 'buffer zone' of the absolute closest path
236            // (i.e. the most probable path)
237            }else if( distance - CLOSEST_PATH_DISTANCE > 
238                            EnvironmentVariables.MAXIMUM_BUFFER_DISTANCE)
239            {
240                    // exponentially increase the cost of a StreetPath based on how much farther
241                    // from a goal segment it is than the closest path
242                    powerFactor = distance / CLOSEST_PATH_DISTANCE;
243            }
244            
245            // make sure the distance factor remains the baseline.
246            return distance * Math.max(distanceFactor, 
247                            Math.pow(distanceFactor, powerFactor));
248        }
249        
250        /**
251         * @throw IllegalStateException if the representation invariant is broken 
252         */
253        private void checkRep(){
254            StreetSegment curr = node;
255            Iterator<StreetSegment> iter = iterator();
256            while(iter.hasNext()){
257                    StreetSegment next = iter.next();
258                    if(!curr.getP1().equals(next.getP2())){
259                            throw new IllegalStateException("Consecutive StreetSegments of this" +
260                                            " StreetPath must have matching points:  this segment's end point" +
261                                            " must match the next segment's start point");
262                    }
263                    curr = next;
264            }
265        }
266        
267        /**
268         * Specified by Path interface.  In addition:
269         * @requires specified StreetSegment is not null.  Also, the start point of 
270         * specified StreetSegment must match end point of this path's end segment.
271         * @throws IllegalArgumentException if passed an invalid StreetSegment
272         */
273            public StreetPath extend(StreetSegment gs){
274                    if(!node.getP2().equals(gs.getP1())){
275                            throw new IllegalArgumentException("cannot extend StreetPath with a segment that" +
276                                            " does not start at this StreetPath's end point");
277                    }
278                    
279                    return new StreetPath(gs, this, goals, start, geographicMap);
280            }

283            /**
284             * Specified by Path interface.  In addition:
285             * StreetPath cost is defined as the total weighted-distance traversed by the path's
286             * StreetSegments so far, plus the geographic distance to the nearest goal segment
287             * (if goal segments were specified).  Additional cost-determination features
288             * available and are controlled by ps6.EnvironmentVariables.java
289             * @returns a double representing the path's cost
290             */
291            public double cost(){
292                    return cost;
293            }
294            
295            
296        /**
297         * Specified by Path interface (which extends Iterable)
298         * Additionally, if fractional segments are activated, the first segment 
299         * returned represents the fractional StreetSegment for the start Address 
300         * of this path.  If the last segment of the path is a goal segment, then 
301         * it too is represented as a fractional StreetSegment for the end address
302         * of that goal segment.
303         */
304            public Iterator<StreetSegment> iterator(){
305            // reverse the linked list, so that elements are returned in order
306            // from start to end of the path.
307            List<StreetSegment> accumulator = new LinkedList<StreetSegment>();
308            
309            if(EnvironmentVariables.FRACTIONAL_SEGMENTS_ON){
310                    accumulator.add(determineFractional(node));
311                    
312                    if(path != null){
313                        for (StreetPath cur = path; cur.path!=null; cur = cur.path) {
314                            accumulator.add(0, cur.end());
315                        }
316                            accumulator.add(0, start.getFractionalSegment());
317                    }
318            }else{
319                for (StreetPath cur = this; cur!=null; cur = cur.path) {
320                    accumulator.add(0, cur.end());
321                }
322            }
323            
324            return accumulator.iterator();
325            }
326            
327            /** helper to determine if a fractional segment should be used to build this route. 
328             * @return the fractional StreetSegment of the specified street if it is a start
329             * or end segment, else returns the street right back */
330            private StreetSegment determineFractional(StreetSegment street){
331                    StreetEntry end = null;
332                    for(StreetEntry se : goals){
333                            if(se.getStreet().equals(street)){
334                                    end = se;
335                                    break;
336                            }
337                    }
338                    if(end != null && start.equals(end)){  // street is part of the end
339                            return start.getSameStreetFractional(end);
340                    }else if(end != null){
341                            return end.getFractionalSegment();
342                    }else if(start.getStreet().equals(street)){
343                            return start.getFractionalSegment();
344                    }
345                    return street;
346            }

353            public StreetSegment end(){
354                    return node;
355            }

359         * @return true iff o is a StreetPath and o.elements is the
360         * same sequence as this.elements

362        @Override
363        public boolean equals(Object o) {
364            if (!(o instanceof StreetPath))
365                return false;
366            return this.equals((StreetPath) o);
367        }

369        /**
370         * @return true iff sp.elements is the same sequence as this.elements
371         **/
372        public boolean equals(StreetPath sp) {
373            return (sp != null) &&
374                this.node.equals(sp.node) && (this.cost() == sp.cost()) &&
375                (this.path == null ? sp.path==null : this.path.equals(sp.path));
376        }

385        /**
386         * Compares the cost of this path to the given path.
387         * @return the value 0 if the cost of this path is equal to the
388         *         cost of the given path; a value less than 0 if this.cost is
389         *         less than p.cost; and a value greater than 0 if this.cost
390         *         is greater than p.cost.
391         * @see java.lang.Comparable#compareTo
392         */
393            public int compareTo(Path<?, ?> sp){
394                    return Double.compare(this.cost(), sp.cost());
395            }
