001    package ps7;
002    
003    /**
004     * Class of static variables and switches that control the behavior of extra
005     * features of HuskyMaps.  Accessed by TextUI and GraphicUI with command line arguments.
006     */
007    
008    public class EnvironmentVariables {
009    
010            /** Switch to turn on fractional segments */
011            public static boolean FRACTIONAL_SEGMENTS_ON =  false;
012    
013            /** Switch to turn on intersection traversal cost */
014            public static boolean INTERSECTION_TRAVERSAL_ON = false;
015            /** Standard measure of the cost of traversing an intersection, in miles. 
016             * Based on an average wait time of ONE MINUTE at each intersection.
017             * ONE MINUTE wait time calculated for an average speed of 2 minutes-per-mile. 
018             * ( 30 MPH , .5 MPminute) */
019            public static final double INTERSECTION_TRAVERSAL_COST = .5;
020            
021            /** Switch to turn on street classification cost */
022            public static boolean STREET_CLASSIFICATION_ON = false;
023            /** Modifier of based on Street Classification, representing standard 
024             * minutes-per-mile averages for primary highway road type. */
025            public static final double PRIM_HWY_COST = 60.0/60.0;
026            /** Modifier of based on Street Classification, representing standard 
027             * minutes-per-mile averages for secondary highway road type. */
028            public static final double SEC_HWY_COST = 60.0/35.0;
029            /** Modifier of based on Street Classification, representing standard 
030             * minutes-per-mile averages for local road type. */
031            public static final double LOCAL_ROAD_COST = 60.0/25.0;
032            /** Modifier of based on Street Classification, representing standard 
033             * minutes-per-mile averages for an unknown road type. */
034            public static final double UNKNOWN_ROAD_COST = 60.0/15.0;
035            
036        /** Factor for modified A* of cost = path length + (distance to nearest goal) * FACTOR 
037         *  Increasing factor improves PathFinding operating time with a corresponding loss in accuracy.
038         *  Meant to be used in conjunction with the MAXIMUM_BUFFER_DISTANCE.  This PathFinding
039         *  optimization will be active if either intersection traversal or street classification is 
040         *  turned on. */
041            public static final double DISTANCE_TO_GOAL_COST_FACTOR = 3.00;
042            /** Maximum buffer distance in miles surrounding the leading edge of the most probable path.
043             * Within this buffer distance, path distance-to-goal cost ( of the A* path finding algorithm )
044             * is determined normally.  Outside this distance, path distance-to-goal cost should be
045             * exponentially increased.  This PathFinding optimization will be active if either 
046             * intersection traversal or street classification is turned on. */
047            public static final double MAXIMUM_BUFFER_DISTANCE = 1.0;
048                    
049            
050            /** Processes the specified command line flag and sets the corresponding switch */
051            public static void setFlag(String flag){
052    
053                    if(flag.equals("-F")){
054                            FRACTIONAL_SEGMENTS_ON =  true;
055                    }else if(flag.equals("-f")){
056                            FRACTIONAL_SEGMENTS_ON =  false;
057                    }else if(flag.equals("-I")){
058                            INTERSECTION_TRAVERSAL_ON = true;
059                    }else if(flag.equals("-i")){
060                            INTERSECTION_TRAVERSAL_ON = false;
061                    }else if(flag.equals("-C")){
062                            STREET_CLASSIFICATION_ON = true;
063                    }else if(flag.equals("-c")){
064                            STREET_CLASSIFICATION_ON = false;
065                    }
066                    
067            }
068            
069    }
