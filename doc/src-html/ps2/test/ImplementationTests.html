056            this("Problem Set 2 ImplementationTests");

059        public ImplementationTests(String name) {
060            super(name);
061            
062            
063            // initialize GeoSegments with value significant in relativity,
064            // descriptions are above field declarations
065            gsEq1 = new GeoSegment("Same St",offset(SEATTLE, 570000, -140000), offset
066                            (SEATTLE, -200000, -1140000));
067            gsEq2 = new GeoSegment("Same St",offset(SEATTLE, 570000, -140000), offset
068                            (SEATTLE, -200000, -1140000));
069            gsDiffPoint = new GeoSegment("Same St",offset(SEATTLE, 240000,-30000), 
070                            offset(SEATTLE, -460000, -530000));
071            gsDiffName = new GeoSegment("Diff Dr",offset(SEATTLE, 570000, -140000), 
072                            offset(SEATTLE, -200000, -1140000));
073            
074            gsAdd1 = new GeoSegment("Same St", offset(SEATTLE, -200000, -1140000), 
075                            offset(SEATTLE, 430000, -360000));
076            gsAdd2 = new GeoSegment("Same St", offset(SEATTLE, 430000, -360000),
077                             offset(SEATTLE, 1200000, 130000));
078            gsAdd3 = new GeoSegment("Same St", offset(SEATTLE, -200000, -1140000),
079                            offset(SEATTLE, -60000, 420000));
080            gsZero = new GeoSegment("Zero zone",offset(SEATTLE, -200000, -1140000), 
081                            offset(SEATTLE, -200000, -1140000));
082            
083    
084            
085            // initialize GeoFeatures
086            gfEq1 = new GeoFeature(gsEq1);
087            gfEq2 = new GeoFeature(gsEq2);
088            gfDiffPoint = new GeoFeature(gsDiffPoint);
089            gfDiffName = new GeoFeature(gsDiffName);
090            gfZero = new GeoFeature(gsZero);
091        }
092    
093        /**
094         * helper to create a new GeoPoint offset from an original
095         * @return a new GeoPoint offset by the specified lat and long
096         */
097        GeoPoint offset(GeoPoint pt, int lat, int lon){
098            return new GeoPoint(pt.getLatitude() + lat, pt.getLongitude() + lon);
099        }
