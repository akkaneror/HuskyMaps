183        /**
184         * helper to create a new GeoPoint offset from an original
185         * @return a new GeoPoint offset by the specified lat and long
186         */
187        GeoPoint offset(GeoPoint pt, int lat, int lon){
188            return new GeoPoint(pt.getLatitude() + lat, pt.getLongitude() + lon);
189        }

192        /**
193         * Convert a degree, minutes, seconds latitude or longitude value
194         * to its associated integer value (to millionths of a degree).
195         * Allows invalid locations to be specified for testing, but
196         * doesn't allow negative minute/second values.
197         **/
198        static int locationToInt(int deg, int min, int sec) {
199            if (min < 0 || sec < 0) {
200                throw new RuntimeException("min and sec must be positive!");
201            }
202    
203            double dmin = (double)min / 60;
204            double dsec = (double)sec / (60 * 60);
205            if (deg >= 0) {
206                return ((int)Math.round(1000000 * (deg + dmin + dsec)));
207            } else {
208                return ((int)Math.round(1000000 * (deg - dmin - dsec)));
209            }
210        }

220            } else if (lng != -71060278) {

235        /**
236         * Tests that a GeoSegment will output the constructor-passed properties.
237         **/
238        public void checkLegalValues() {
239            GeoPoint pt1 = offset(SEATTLE, 2000, -7000);
240            GeoPoint pt2 = offset(SEATTLE, 300, 11200);
241            
242            GeoSegment gs = null;
243            String name = "Segment Rd";
244            try {
245                gs = new GeoSegment(name, pt1, pt2);
246            } catch (Exception ex) {
247                // Failed
248                fail("Didn't allow legal (start P1, end P2) of (" +
249                            pt1 + ", " + pt2 + ")");
250                return;
251            }
252    
253            if (!gs.getName().equals(name)){
254                    fail("Name should have been set to " + name + ", was set to "+
255                                    gs.getName());
256            }else if (gs.getLength() != pt1.distanceTo(pt2)) {
257                fail("Length should have been set to " + pt1.distanceTo(pt2) +
258                            ", was set to " + gs.getLength() + " instead.");
259    
260            }else if (gs.getHeading() != pt1.headingTo(pt2)) {
261                fail("Longitude should have been set to " + pt1.headingTo(pt2) +
262                            ", was set to " + gs.getHeading() + " instead.");
263            }
264        }

270        /**
271         * test positive reverse: the headings and distances of the four prime directions
272         */
273        public void testReverseCompassDirections(){
274            
275            // true directions
276            testReverse(gsNorth, nDist, nHeading);
277            testReverse(gsSouth, sDist, sHeading);
278            testReverse(gsEast, eDist, eHeading);
279            testReverse(gsWest, wDist, wHeading);
280            assertTrue("A reversed north is equal to south", gsNorth.reverse().
281                            getHeading() == gsSouth.getHeading());
282            assertTrue("A reversed east is equal to west", gsEast.reverse().
283                            getHeading() == gsWest.getHeading());
284            
285            // diagonal directions
286            testReverse(gsNorthEast, neDist, neHeading);
287            testReverse(gsNorthWest, nwDist, nwHeading);
288            testReverse(gsSouthEast, seDist, seHeading);
289            testReverse(gsSouthWest, swDist, swHeading);
290            assertTrue("A reversed northEast will point southWest", gsNorthEast.
291                            reverse().getHeading() == gsSouthWest.getHeading());
292            assertTrue("A reversed northWest will point soutEast", gsNorthWest.
293                            reverse().getHeading() == gsSouthEast.getHeading());
294        }

304        /**
305         * helper to test properties of reversed segments
306         */
307        void testReverse(GeoSegment gs, double len, double hea){
308            GeoSegment rev = gs.reverse();
309            if(hea - 180 < 0){
310                    hea += 180;
311            }else{
312                    hea -= 180;
313            }
314            
315            assertEquals("A reversed segment must have the same length", 
316                            len, rev.getLength(), TOLERANCE);
317            assertEquals("A reversed segment must have the opposite heading",
318                            hea, rev.getHeading(), TOLERANCE);
319            assertTrue("A reversed segment should not be equal to itself",
320                            !rev.equals(gs));
321            assertTrue("A double reversed segment should be equal to itself",
322                            rev.reverse().equals(gs));
323        }

387         * Helper to check the length of a specified GeoSegment
388         */
389        void checkLength(GeoSegment gs, double dis){
390            assertEquals("Length of segment is "+dis+", not "+
391                            gs.getLength(), dis, gs.getLength(), TOLERANCE);
392            double man_dist = gs.getP1().distanceTo(gs.getP2());
393            assertEquals("Length of segment is "+dis+", not "+
394                            man_dist, dis, man_dist, TOLERANCE);
395            assertEquals("getPx() and getLength() methods do not agree",
396                            man_dist, gs.getLength(), TOLERANCE);
397        }
