189        /**
190         * helper to create a new GeoPoint offset from an original
191         * @return a new GeoPoint offset by the specified lat and long
192         */
193        GeoPoint offset(GeoPoint pt, int lat, int lon){
194            return new GeoPoint(pt.getLatitude() + lat, pt.getLongitude() + lon);

382        /**
383         * test getGeoFeatures 
384         */
385        public void testGetFeatures(){
386            int i = 0;
387            Route r = new Route(gsList.get(i));
388            while(i < gsList.size() - 1){
389                    i++;
390                    r = r.addSegment(gsList.get(i));
391            }
392            
393            List<GeoFeature> a = r.getGeoFeatures();
394            // this.start        = a.get(0).p1
395            assertTrue("The start point for the route should be the first" +
396                            " point of the first element of the returned list", 
397                            r.getStart().equals(gfList.get(0).getStart()));
398            // this.startHeading = a.get(0).heading
399            testStartHeading(gsList.get(0), r); 
400            // this.end = a.get(a.size - 1).p2
401            assertTrue("The end point should be the second point of the" +
402                            " last element of the list", r.getEnd().equals
403                            (gfList.get(gfList.size() - 1).getEnd()));
404            // this.endHeading = a.get(a.size - 1).heading
405            testEndHeading(gsList.get(gsList.size() - 1), r);
406            // this.length =  sum (0 & le; i &lt; a.size) . a.get(i).length
407            double sum = 0;
408            for(GeoFeature gf : a){
409                    sum += gf.getLength();
410            }
411            assertEquals("The total length of the features contained in the list" +
412                            " should equal the length of the Route", r.getLength(),
413                            sum, TOLERANCE);
414            
415            
416            i = 0;
417            while(i < a.size()){
418                    if(!gfList.get(i).equals(a.get(i))){
419                                    System.out.println(gfList.get(i) +"=========VS========"+(a.get(i)));
420                            fail("The segments contained should be organized into features "+
421                            "of continuous segments");
422                    }
423                    i++;
424            }
425        }

520        /**
521         * test that unequal segment order equates to unequal features
522         */
523        public void testUnequalSegmentOrder(){
524            Route r1 = new Route(gsEq1);
525            Route r2 = new Route(gsEq1);
526    
527            
528            //test segment order
529            r1 = r1.addSegment(gsAdd3);
530            r1 = r1.addSegment(gsAdd3.reverse());
531            r1 = r1.addSegment(gsAdd1);
532            r1 = r1.addSegment(gsAdd1.reverse());
533            r1 = r1.addSegment(gsAdd1);
534            
535            r2 = r2.addSegment(gsAdd1);
536            r2 = r2.addSegment(gsAdd1.reverse());
537            r2 = r2.addSegment(gsAdd3);
538            r2 = r2.addSegment(gsAdd3.reverse());
539            r2 = r2.addSegment(gsAdd1);
540            
541            assertTrue("Routes of equal properties, but different segment order" +
542                            "should not be equal", !r1.equals(r2));
543            
544            
545            // again with zero-length segments the only difference
546            r1 = new Route(gsEq1);
547            r2 = new Route(gsEq1);
548            GeoSegment z = new GeoSegment(gsAdd1.getName(), gsAdd1.getP1(),
549                            gsAdd1.getP1());
550            
551            r1 = r1.addSegment(z);
552            r1 = r1.addSegment(gsAdd1);
553            r1 = r1.addSegment(gsAdd1.reverse());
554            r1 = r1.addSegment(gsAdd1);
555            
556            r2 = r2.addSegment(gsAdd1);
557            r2 = r2.addSegment(gsAdd1.reverse());
558            r2 = r2.addSegment(z);
559            r2 = r2.addSegment(gsAdd1);
560            
561            assertTrue("Routes of equal properties, but different segment order" +
562                            "should not be equal", !r1.equals(r2));
563        }
