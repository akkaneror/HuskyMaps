165         * GeoFeature constructor @requires gs != null
166         */
167        public void testLegalValues(){
168            
169            GeoFeature gf = null;
170            try {
171                gf = new GeoFeature(gsEq1);
172            } catch (Exception ex) {

174                fail("Didn't allow legal (GeoSegment) construction from "+gsEq1);
175                return;
176            }

179                    fail("feature name should have been set to "+gsEq1.getName()+
180                                    ", but was set to "+gf.getName()+" instead");
181            }else if(!gf.getStart().equals(gsEq1.getP1())){
182                    fail("feature start should have been set to "+gsEq1.getP1()+
183                                    ", but was set to "+gf.getStart()+" instead");
184            }else if(!gf.getEnd().equals(gsEq1.getP2())){
185                    fail("feature start should have been set to "+gsEq1.getP1()+
186                                    ", but was set to "+gf.getEnd()+" instead");
187            }
188        }

513        /**
514         * test that unequal segment order equates to unequal features
515         */
516        public void testUnequalSegmentOrder(){
517            GeoFeature gf1 = new GeoFeature(gsEq1);
518            GeoFeature gf2 = new GeoFeature(gsEq1);
519    
520            
521            //test segment order
522            gf1 = gf1.addSegment(gsAdd3);
523            gf1 = gf1.addSegment(gsAdd3.reverse());
524            gf1 = gf1.addSegment(gsAdd1);
525            gf1 = gf1.addSegment(gsAdd1.reverse());
526            gf1 = gf1.addSegment(gsAdd1);
527            
528            gf2 = gf2.addSegment(gsAdd1);
529            gf2 = gf2.addSegment(gsAdd1.reverse());
530            gf2 = gf2.addSegment(gsAdd3);
531            gf2 = gf2.addSegment(gsAdd3.reverse());
532            gf2 = gf2.addSegment(gsAdd1);
533            
534            assertTrue("Features of equal properties, but different segment order" +
535                            "should not be equal", !gf1.equals(gf2));
536            
537            
538            GeoSegment z = new GeoSegment(gsAdd1.getName(), gsAdd1.getP1(),
539                            gsAdd1.getP1());
540            // again with zero-length segments the only difference
541            gf1 = new GeoFeature(gsEq1);
542            gf2 = new GeoFeature(gsEq1);
543            
544            gf1 = gf1.addSegment(z);
545            gf1 = gf1.addSegment(gsAdd1);
546            gf1 = gf1.addSegment(gsAdd1.reverse());
547            gf1 = gf1.addSegment(gsAdd1);
548            
549            gf2 = gf2.addSegment(gsAdd1);
550            gf2 = gf2.addSegment(gsAdd1.reverse());
551            gf2 = gf2.addSegment(z);
552            gf2 = gf2.addSegment(gsAdd1);
553            
554            assertTrue("Features of equal properties, but different segment order" +
555                            "should not be equal", !gf1.equals(gf2));
556        }
