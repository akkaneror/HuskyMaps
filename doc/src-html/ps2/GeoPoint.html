001    package ps2;
002    
003    
004    /**
005     * <p>
006     * A GeoPoint models a point on the earth.  GeoPoints are immutable.
007     * </p>
008     *
009     * <p>
010     * North latitudes and east longitudes are represented by positive
011     * numbers.  South latitudes and west longitudes are represented by
012     * negative numbers.
013     * </p>
014     *
015     * <p>
016     * The code may assume that the represented points are nearby Seattle.
017     * </p>
018     *
019     * <p>
020     * <b>Implementation hint</b>:<br /> Seattle is at approximately 47
021     * deg. 36 min. 35 sec. N latitude and 122 deg. 19 min. 59 sec. W
022     * longitude.  There are 60 minutes per degree, and 60 seconds per
023     * minute.  So, in decimal, these correspond to 47.609722 North
024     * latitude and -122.333056 East longitude.  The constructor takes
025     * integers in millionths of degrees.  To create a new GeoPoint
026     * located in Seattle, use: <tt>GeoPoint seattle = new
027     * GeoPoint(47609722, -122333056);</tt>
028     * </p>
029     *
030     * <p>
031     * Near Seattle, there are approximately 69.04 miles per degree of
032     * latitude and 47.574 miles per degree of longitude.  An
033     * implementation should use these values when determining distances
034     * and headings.
035     * </p>
036     * 
037     *
038     * @specfield  latitude :  real     // measured in degrees latitude
039     * @specfield  longitude : real     // measured in degrees longitude
040     **/
041    public strictfp class GeoPoint {
042    
043             /* 
044             * ABSTRACTION FUNCTION
045             * latitude  >>  the latitude in millionths of degrees of this location
046             * longitude  >>  the longitude in millionths of degrees of this location
047             * 
048             * REPRESENTATION INVARIANT
049             * -90000000 < latitude < 90000000
050             * -180000000 < longitude < 180000000
051             */ 
052            
053            
054        /** Minimum value the latitude field can have in this class. **/
055        public static final int MIN_LATITUDE  =  -90 * 1000000;
056        /** Maximum value the latitude field can have in this class. **/
057        public static final int MAX_LATITUDE  =   90 * 1000000;
058        /** Minimum value the longitude field can have in this class. **/
059        public static final int MIN_LONGITUDE = -180 * 1000000;
060        /** Maximum value the longitude field can have in this class. **/
061        public static final int MAX_LONGITUDE =  180 * 1000000;
062    
063    
064        /**
065         * Approximation used to determine distances and headings using a
066         * "flat earth" simplification.
067         **/
068        public static final double MILES_PER_DEGREE_LATITUDE = 69.04;
069    
070        /**
071         * Approximation used to determine distances and headings using a
072         * "flat earth" simplification.
073         **/
074        public static final double MILES_PER_DEGREE_LONGITUDE = 46.574;
075    
076    
077        // FIELDS
078        
079        /**
080         * This GeoPoint's latitude in millionths of degrees
081         */
082        private final int latitude;
083        
084        /**
085         * This GeoPoint's longitude in millionths of degrees
086         */
087        private final int longitude;
088    
089        /**The hash code for this point*/
090        private final int hash;
091        
092        // Constructors
093    
094        /**
095         * @requires the (latitude, longitude) point expressed in
096         * millionths of a degree is valid, such that
097         * MIN_LATITUDE <= latitude <= MAX_LATITUDE and
098         * MIN_LONGITUDE <= longitude <= MAX_LONGITUDE
099         * 
100         * @effects constructs a GeoPoint from a latitude and
101         * longitude given in millionths of degrees.
102         **/
103        public GeoPoint(int latitude, int longitude) {
104            this.latitude = latitude;
105            this.longitude = longitude;
106            hash = Math.abs(latitude - longitude);
107            checkRep();
108        }
109    
110    
111        /**
112         * Checks that the representation invariant holds (if any).
113         * @throws RuntimeException if the rep invariant is violated.
114             **/
115        private void checkRep() {
116                    if(MIN_LATITUDE > latitude || latitude > MAX_LATITUDE){
117                            throw new RuntimeException("invalid latitude");
118                    }else if(MIN_LONGITUDE > longitude || longitude > MAX_LONGITUDE){
119                            throw new RuntimeException("invalid longitude");
120                    }
121        }
122    
123    
124        // Observers
125    
126        /**
127         * the latitude of the GeoPoint object, in millionths of degrees.
128         **/
129        public int getLatitude() {
130            return latitude;
131        }
132    
133    
134        /**
135         * the longitude of the GeoPoint object, in millionths of degrees.
136         **/
137        public int getLongitude() {
138            return longitude;
139        }
140    
141    
142        /**
143         * Computes the distance between GeoPoints.
144         * @requires gp != null
145         * @throws IllegalArgumentException is gp == null
146         * @return The distance, in miles, from this to gp, using the
147         * flat-surface, near Seattle approximation.
148         **/
149        public double distanceTo(GeoPoint gp) {
150            if(gp == null){
151                    throw new IllegalArgumentException("specified GeoPoint gp cannot be null");
152            }
153            double lat = (gp.getLatitude() - latitude) * MILES_PER_DEGREE_LATITUDE /
154                            1000000;
155            double lon = (gp.getLongitude() - longitude) * MILES_PER_DEGREE_LONGITUDE /
156                            1000000;
157            return Math.sqrt(Math.pow(lat, 2) + Math.pow(lon, 2)); 
158        }
159    
160    
161        /**
162         * Computes the compass heading between GeoPoints.
163         * @requires gp != null && !this.equals(gp)
164         * @throws IllegalArgumentException if gp == null
165         * @return The compass heading h from this to
166         * gp, in degrees, using the flat-surface, near Seattle
167         * approximation, such that 0 &le; h &lt; 360.  In compass
168         * headings, north = 0, east = 90, south = 180, and west = 270.
169         **/
170        // If gp is at the same GeoPoint as this, returns Double.NaN
171        public double headingTo(GeoPoint gp) {
172            if(gp == null){
173                    throw new IllegalArgumentException("specified GeoPoint " +
174                                    "cannot be null");
175            }else if(this.equals(gp)){
176                    // if this is the same point as gp, returns Double.NaN
177                    return Double.NaN;
178            }
179            
180            double delLat = (gp.getLatitude() - latitude) * MILES_PER_DEGREE_LATITUDE /
181                            1000000;
182            double delLon = (gp.getLongitude() - longitude) * MILES_PER_DEGREE_LONGITUDE /
183                            1000000;
184            double deg = Math.toDegrees(Math.atan2(delLat, delLon));
185            
186            deg = 90 - deg;
187            if(deg < 0){
188                    deg = deg + 360;
189            }
190            return deg;
191        }
192    
193        /**
194         * Compares the specified Object with this GeoPoint for equality.
195         * @return gp != null && (gp instanceof GeoPoint)
196         *         && gp.latitude = this.latitude && gp.longitude = this.longitude
197         **/
198        public boolean equals(Object gp) {
199            if(gp == null || hashCode() != gp.hashCode()){
200                    return false;
201            }
202            return ((GeoPoint) gp).getLatitude() == latitude && ((GeoPoint) gp).
203                            getLongitude() == longitude;
204        }
205    
206    
207        /**
208         * @return a valid hashcode for this GeoPoint.
209         **/
210        public int hashCode() {
211            return Math.abs(latitude * longitude);
212        }
213    
214    
215        /**
216         * @return a string representation of this GeoPoint.
217         * representation is of the form: "(latitude, longitude)" .
218         **/
219        public String toString() {
220            return "("+latitude+", "+longitude+")";
221        }
222    
223        
224    } // GeoPoint
