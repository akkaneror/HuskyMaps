008    /**
009     * <b>RatPoly</b> represents an immutable single-variate polynomial expression.
010     * RatPolys are sums of RatTerms with non-negative exponents.
011     * <p>
012     *
013     * Examples of RatPolys include "0", "x-10", and "x^3-2*x^2+5/3*x+3", and "NaN".
014     */
015    // See RatNum's documentation for a definition of "immutable".
016    public final class RatPoly {
017    
018        /** Holds all the RatTerms in this RatPoly */
019        private final List<RatTerm> terms;
020    
021        // Definitions:
022        // For a RatPoly p, let C(p,i) be "p.terms.get(i).getCoeff()" and
023        // E(p,i) be "p.terms.get(i).getExpt()"
024        // length(p) be "p.terms.size()"
025        // (These are helper functions that will make it easier for us
026        // to write the remainder of the specifications. They are not
027        // executable code; they just represent complex expressions in a
028        // concise manner, so that we can stress the important parts of
029        // other expressions in the spec rather than get bogged down in
030        // the details of how we extract the coefficient for the 2nd term
031        // or the exponent for the 5th term. So when you see C(p,i),
032        // think "coefficient for the ith term in p".)
033        //
034        // Abstraction Function:
035        // RatPoly, p, represents the polynomial equal to the sum of the
036        // RatTerms contained in 'terms':
037        // sum (0 <= i < length(p)): p.terms.get(i)
038        // If there are no terms, then the RatPoly represents the zero
039        // polynomial.
040        //
041        // Representation Invariant for every RatPoly p:
042        // terms != null &&
043        // forall i such that (0 <= i < length(p)), C(p,i) != 0 &&
044        // forall i such that (0 <= i < length(p)), E(p,i) >= 0 &&
045        // forall i such that (0 <= i < length(p) - 1), E(p,i) > E(p, i+1)
046        // In other words:
047        // * The terms field always points to some usable object.
048        // * No term in a RatPoly has a zero coefficient.
049        // * No term in a RatPoly has a negative exponent.
050        // * The terms in a RatPoly are sorted in descending exponent order.
051        // (It is implied that 'terms' does not contain any null elements by the
052        // above
053        // invariant.)
054    
055        /** A constant holding a Not-a-Number (NaN) value of type RatPoly */
056        public static final RatPoly NaN = new RatPoly(RatTerm.NaN);
057    
058        /** A constant holding a zero value of type RatNum */
059        public static final RatPoly ZERO = new RatPoly();

098            terms = rt;
099            // The spec tells us that we don't need to make a copy of 'rt'
100            checkRep();

104         * Returns the degree of this RatPoly.
105         *
106         * @requires !this.isNaN()
107         * @return the largest exponent with a non-zero coefficient, or 0 if this is
108         *         "0".
109         */
110        public int degree() {
111            if(terms.size() == 0){
112                    return 0;
113            }
114            return terms.get(0).getExpt();

121         * @return the RatTerm of degree 'deg'. If there is no term of degree 'deg'
122         *         in this poly, then returns the zero RatTerm.
123         */
124        public RatTerm getTerm(int deg) {
125            for(RatTerm rt: terms){
126                    if(rt.getExpt() == deg){
127                            return rt;
128                    }
129            }
130            return RatTerm.ZERO;
131        }

139            for(RatTerm rt: terms){
140                    if(rt.isNaN()){

148         * Scales coefficients within 'lst' by 'scalar' (helper procedure).
149         *
150         * @requires lst, scalar != null
151         * @modifies lst
152         * @effects Forall i s.t. 0 <= i < lst.size(), if lst.get(i) = (C . E) then
153         *          lst_post.get(i) = (C*scalar . E)
154         * @see ps1.RatTerm regarding (C . E) notation
155         */
156        private static void scaleCoeff(List<RatTerm> lst, RatNum scalar) {
157                    for(int i = 0; i < lst.size(); i++){
158                    lst.set(i, lst.get(i).mul(new RatTerm(scalar, 0)));
159            }
160        }

162        /**
163         * Increments exponents within 'lst' by 'degree' (helper procedure).
164         *
165         * @requires lst != null
166         * @modifies lst
167         * @effects Forall i s.t. 0 <= i < lst.size(), if (C . E) = lst.get(i) then
168         *          lst_post.get(i) = (C . E+degree)
169         * @see ps1.RatTerm regarding (C . E) notation
170         */
171        private static void incremExpt(List<RatTerm> lst, int degree) {
172            for(int i = 0; i < lst.size(); i++){
173                    lst.set(i, lst.get(i).mul(new RatTerm(new RatNum(1), degree)));
174                    if(lst.get(i).getExpt() < 0){
175                            lst.remove(i);
176                    }
177            }
178        }

180        /**
181         * Helper procedure: Inserts a term into a sorted sequence of terms,
182         * preserving the sorted nature of the sequence. If a term with the given
183         * degree already exists, adds their coefficients.
184         *
185         * Definitions: Let a "Sorted List<RatTerm>" be a List<RatTerm> V such
186         * that [1] V is sorted in descending exponent order && [2] there are no two
187         * RatTerms with the same exponent in V && [3] there is no RatTerm in V with
188         * a coefficient equal to zero
189         *
190         * For a Sorted List<RatTerm> V and integer e, let cofind(V, e) be either
191         * the coefficient for a RatTerm rt in V w          int b = 0;hose exponent is e, or zero if
192         * there does not exist any such RatTerm in V. (This is like the coeff
193         * function of RatPoly.) We will write sorted(lst) to denote that lst is a
194         * Sorted List<RatTerm>, as defined above.
195         *
196         * @requires lst != null && sorted(lst)
197         * @modifies lst
198         * @effects sorted(lst_post) && (cofind(lst_post,newTerm.getExpt()) =
199         *          cofind(lst,newTerm.getExpt()) + newTerm.getCoeff())
200         */
201        private static void sortedInsert(List<RatTerm> lst, RatTerm newTerm) {
202            if(!newTerm.getCoeff().equals(RatNum.ZERO)){
203                    if(!lst.isEmpty()){
204                            int i = 0;
205                            
206                            boolean added = false;
207                            while(i < lst.size() && !added){ // iterate over all elements
208    
209                                    if(lst.get(i).getExpt() == newTerm.getExpt()){ //if degrees are equal
210                                            
211                                            if(lst.get(i).add(newTerm).getCoeff().equals(RatNum.ZERO)){ //if sum == 0
212                                                    lst.remove(i);
213                                            }else{ // set to temp
214                                                    lst.set(i, lst.get(i).add(newTerm));
215                                            }
216                                            added = true;
217                                            
218                                    }else if(lst.get(i).getExpt() < newTerm.getExpt()){
219                                            lst.add(i, newTerm);    
220                                            added = true;
221                                    }
222    
223                                    i++;
224                            }
225                            if(!added){
226                                    lst.add(newTerm);
227                            }
228                            
229                        }else{
230                            lst.add(newTerm);
231                    }
232            }
233        }

238         * @return a RatPoly equal to "0 - this"; if this.isNaN(), returns some r

244        public RatPoly negate() {
245            List<RatTerm> list = new ArrayList<RatTerm>();
246            for(RatTerm rt: terms){
247                    list.add(rt);
248            }
249            scaleCoeff(list, new RatNum(-1));

253        /**
254         * Addition operation.
255         *
256         * @requires p != null
257         * @return a RatPoly, r, such that r = "this + p"; if this.isNaN() or
258         *         p.isNaN(), returns some r such that r.isNaN()
259         */
260        public RatPoly add(RatPoly p) {
261            if(this.isNaN() || p.isNaN()){
262                    return NaN;
263            }
264            
265            List<RatTerm> list = new ArrayList<RatTerm>();
266            int i = 0;
267            while(i <= degree() || i <= p.degree()){
268                    sortedInsert(list, getTerm(i));
269                    sortedInsert(list, p.getTerm(i));
270                    i++;
271            }
272            return new RatPoly(list);
273        }

276         * Subtraction operation.
277         *
278         * @requires p != null
279         * @return a RatPoly, r, such that r = "this - p"; if this.isNaN() or
280         *         p.isNaN(), returns some r such that r.isNaN()
281         */
282        public RatPoly sub(RatPoly p) {
283            return add(p.negate());
284        }

286        /**
287         * Multiplication operation.
288         *
289         * @requires p != null
290         * @return a RatPoly, r, such that r = "this * p"; if this.isNaN() or
291         *         p.isNaN(), returns some r such that r.isNaN()
292         */
293        public RatPoly mul(RatPoly p) {
294            if(this.isNaN() || p.isNaN()){
295                    return NaN;
296            }
297            
298            List<RatTerm> list = new ArrayList<RatTerm>();
299            int i = 0;
300            int j = 0;
301            while(i <= degree()){ // for all terms in this.list
302                    if(j <= p.degree()){ // iterate through all terms of p.list
303                            
304                            // take two terms and multiply together.  Add result to list
305                            sortedInsert(list, getTerm(i).mul(p.getTerm(j)));
306                            j++;
307                    }else{ // reset for next this.list term
308                            j = 0;
309                            i++;
310                    }
311            }
312            return new RatPoly(list);
313        }

315        /**
316         * Division operation (truncating).
317         *
318         * @requires p != null
319         * @return a RatPoly, q, such that q = "this / p"; if p = 0 or this.isNaN()
320         *         or p.isNaN(), returns some q such that q.isNaN()
321         *         <p>
322         *
323         * Division of polynomials is defined as follows: Given two polynomials u
324         * and v, with v != "0", we can divide u by v to obtain a quotient
325         * polynomial q and a remainder polynomial r satisfying the condition u = "q *
326         * v + r", where the degree of r is strictly less than the degree of v, the
327         * degree of q is no greater than the degree of u, and r and q have no
328         * negative exponents.
329         * <p>
330         *
331         * For the purposes of this class, the operation "u / v" returns q as
332         * defined above.
333         * <p>
334         *
335         * The following are examples of div's behavior: "x^3-2*x+3" / "3*x^2" =
336         * "1/3*x" (with r = "-2*x+3"). "x^2+2*x+15 / 2*x^3" = "0" (with r =
337         * "x^2+2*x+15"). "x^3+x-1 / x+1 = x^2-x+2 (with r = "-3").
338         * <p>
339         *
340         * Note that this truncating behavior is similar to the behavior of integer
341         * division on computers.
342         */
343        public RatPoly div(RatPoly p) {
344            if(this.isNaN() || p.isNaN() || p.equals(ZERO)){
345                    return NaN;
346            }
347            
348            List<RatTerm> list = new ArrayList<RatTerm>();
349            
350            if(!this.equals(ZERO)){
351                    RatPoly remain = new RatPoly(terms);
352                    
353                    while(remain.degree() >= p.degree() && remain.terms.size() != 0){ // while degree of this is greater than divisor
354            
355                            //set temp to quotient to highest degree term of remainder divided by h_d_t of dividend
356                                    RatTerm temp = remain.getTerm(remain.degree()).div(p.getTerm(p.degree()));
357                                    //check if degree is less than 0
358                                    if(temp.getExpt() >= 0){ //if not, add to list
359            
360                                            //modify r by subtracting (v multiplied by coe and with each term's degree increased by diff)
361                                            remain = remain.sub(p.mul(new RatPoly(temp)));
362                                    
363                                            //modify q by adding a term with coefficient coe and of degree diff 
364                                            sortedInsert(list, temp);
365                            }
366                    }
367            }
368            return new RatPoly(list);
369        }

371        /**
372         * Return the derivative of this RatPoly.
373         *
374         * @return a RatPoly, q, such that q = dy/dx, where this == y. In other
375         *         words, q is the derivative of this. If this.isNaN(), then return
376         *         some q such that q.isNaN()
377         *
378         * <p>
379         * The derivative of a polynomial is the sum of the derivative of each term.
380         */
381        public RatPoly differentiate() {
382            if(this.isNaN()){
383                    return NaN;
384            }
385            
386            List<RatTerm> list = new ArrayList<RatTerm>();
387            for(RatTerm rt: terms){
388                    list.add(rt);
389            }
390            
391            if(!list.isEmpty()){
392                    //remove any zero-degree term
393                    if(list.get(list.size()-1).getExpt() == 0){
394                            list.remove(list.size()-1);
395                    }
396                    
397                    for(int i = 0; i < list.size(); i++){
398                            list.set(i, list.get(i).differentiate());
399                    }
400            }
401            
402            return new RatPoly(list);
403        }

405        /**
406         * Returns the antiderivative of this RatPoly.
407         *
408         * @requires integrationConstant != null
409         * @return a RatPoly, q, such that dq/dx = this and the constant of
410         *         integration is "integrationConstant" In other words, q is the
411         *         antiderivative of this. If this.isNaN() or
412         *         integrationConstant.isNaN(), then return some q such that
413         *         q.isNaN()
414         *
415         * <p>
416         * The antiderivative of a polynomial is the sum of the antiderivative of
417         * each term plus some constant.
418         */
419        public RatPoly antiDifferentiate(RatNum integrationConstant) {
420            if(this.isNaN()){
421                    return NaN;
422            }
423            
424            List<RatTerm> list = new ArrayList<RatTerm>();
425            for(RatTerm rt: terms){
426                    list.add(rt);
427            }
428            
429            for(int i = 0; i < list.size(); i++){
430                    list.set(i, list.get(i).antiDifferentiate());
431            }
432            // add integration constant as a zero-degree term
433            sortedInsert(list, new RatTerm(integrationConstant, 0));
434            return new RatPoly(list);
435        }

437        /**
438         * Returns the integral of this RatPoly, integrated from lowerBound to
439         * upperBound.
440         *
441         * <p>
442         * The Fundamental Theorem of Calculus states that the definite integral of
443         * f(x) with bounds a to b is F(b) - F(a) where dF/dx = f(x) NOTE: Remember
444         * that the lowerBound can be higher than the upperBound.
445         *
446         * @return a double that is the definite integral of this with bounds of
447         *         integration between lowerBound and upperBound. If this.isNaN(),
448         *         or either lowerBound or upperBound is Double.NaN, return
449         *         Double.NaN.
450         */
451        public double integrate(double lowerBound, double upperBound) {
452            if(this.isNaN() || upperBound == Double.NaN || lowerBound == Double.NaN){
453                    return Double.NaN;
454            }
455            
456            List<RatTerm> list = new ArrayList<RatTerm>();
457            
458            for(RatTerm rt: terms){
459                    sortedInsert(list, rt.div(new RatTerm(new RatNum(rt.getExpt()+1), 0)));// multiply each term by it's coefficient
460            }
461            incremExpt(list, 1);
462            
463            return (new RatPoly(list).eval(upperBound) - new RatPoly(list).eval(lowerBound));
464        }

467         * Returns the value of this RatPoly, evaluated at d.
468         *
469         * @return the value of this polynomial when evaluated at 'd'. For example,
470         *         "x+2" evaluated at 3 is 5, and "x^2-x" evaluated at 3 is 6. if

473        public double eval(double d) {
474            if(this.isNaN()){
475                    return Double.NaN;
476            }
477            
478            double sum = 0;
479            for(RatTerm rt: terms){
480                    sum += rt.eval(d);
481            }
482            return sum;

485        /**
486         * Returns a string representation of this RatPoly.
487         *
488         * @return A String representation of the expression represented by this,
489         *         with the terms sorted in order of degree from highest to lowest.
490         *         <p>
491         *         There is no whitespace in the returned string.
492         *         <p>
493         *         If the polynomial is itself zero, the returned string will just
494         *         be "0".
495         *         <p>
496         *         If this.isNaN(), then the returned string will be just "NaN"
497         *         <p>
498         *         The string for a non-zero, non-NaN poly is in the form
499         *         "(-)T(+|-)T(+|-)...", where "(-)" refers to a possible minus
500         *         sign, if needed, and "(+|-)" refer to either a plus or minus
501         *         sign, as needed. For each term, T takes the form "C*x^E" or "C*x"
502         *         where C > 0, UNLESS: (1) the exponent E is zero, in which case T
503         *         takes the form "C", or (2) the coefficient C is one, in which
504         *         case T takes the form "x^E" or "x". In cases were both (1) and
505         *         (2) apply, (1) is used.
506         *         <p>
507         *         Valid example outputs include "x^17-3/2*x^2+1", "-x+1", "-1/2",
508         *         and "0".
509         *         <p>
510         */
511        @Override
512        public String toString() {
513            if (terms.size() == 0) {
514                return "0";
515            }
516            if (isNaN()) {
517                return "NaN";
518            }
519            StringBuilder output = new StringBuilder();
520            boolean isFirst = true;
521            for (RatTerm rt : terms) {
522                if (isFirst) {
523                    isFirst = false;
524                    output.append(rt.toString());
525                } else {
526                    if (rt.getCoeff().isNegative()) {
527                        output.append(rt.toString());
528                    } else {
529                        output.append("+" + rt.toString());
530                    }
531                }
532            }
533            return output.toString();
534        }

536        /**
537         * Builds a new RatPoly, given a descriptive String.
538         *
539         * @requires 'polyStr' is an instance of a string with no spaces that
540         *           expresses a poly in the form defined in the toString() method.
541         *           <p>
542         *
543         * Valid inputs include "0", "x-10", and "x^3-2*x^2+5/3*x+3", and "NaN".
544         *
545         * @return a RatPoly p such that p.toString() = polyStr
546         */
547        public static RatPoly valueOf(String polyStr) {
548    
549            List<RatTerm> parsedTerms = new ArrayList<RatTerm>();
550    
551            // First we decompose the polyStr into its component terms;
552            // third arg orders "+" and "-" to be returned as tokens.
553            StringTokenizer termStrings = new StringTokenizer(polyStr, "+-", true);
554    
555            boolean nextTermIsNegative = false;
556            while (termStrings.hasMoreTokens()) {
557                String termToken = termStrings.nextToken();
558    
559                if (termToken.equals("-")) {
560                    nextTermIsNegative = true;
561                } else if (termToken.equals("+")) {
562                    nextTermIsNegative = false;
563                } else {
564                    // Not "+" or "-"; must be a term
565                    RatTerm term = RatTerm.valueOf(termToken);
566    
567                    // at this point, coeff and expt are initialized.
568                    // Need to fix coeff if it was preceeded by a '-'
569                    if (nextTermIsNegative) {
570                        term = term.negate();
571                    }
572    
573                    // accumulate terms of polynomial in 'parsedTerms'
574                    sortedInsert(parsedTerms, term);
575                }
576            }
577            return new RatPoly(parsedTerms);
578        }

594        /**
595         * Standard equality operation.
596         *
597         * @return true if and only if 'obj' is an instance of a RatPoly and 'this'
598         *         and 'obj' represent the same rational polynomial. Note that all
599         *         NaN RatPolys are equal.
600         */
601        @Override
602        public boolean equals(Object obj) {
603            if (obj instanceof RatPoly) {
604                RatPoly rp = (RatPoly) obj;
605    
606                // special case: check if both are NaN
607                if (this.isNaN() && rp.isNaN()) {
608                    return true;
609                } else {
610                    return terms.equals(rp.terms);
611                }
612            } else {
613                return false;
614            }

617        /**
618         * Checks that the representation invariant holds (if any).
619         */
620        // Throws a RuntimeException if the rep invariant is violated.
621        private void checkRep() throws RuntimeException {
622            if (terms == null) {
623                assert (false) : "CheckRep assert";
624                throw new RuntimeException("terms == null!");
625            }
626            for (int i = 0; i < terms.size(); i++) {
627                if (terms.get(i).getCoeff().equals(new RatNum(0))) {
628                    throw new RuntimeException("zero coefficient!");
629                }
630                if (terms.get(i).getExpt() < 0) {
631                    throw new RuntimeException("negative exponent!");
632                }
633                if (i < terms.size() - 1) {
634                    if (terms.get(i + 1).getExpt() >= terms.get(i).getExpt()) {
635                        throw new RuntimeException("terms out of order!");
636                    }
637                }
638            }
639        }
