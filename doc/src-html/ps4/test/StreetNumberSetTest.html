105            /**
106             * helper to populate the specified array with random numbers with domain 
107             * covering the maximum limits of primitive type int (0 to 1000000).
108             * @return string representation of selected numbers to be used in 
109             * StreetNumberSet constructor. 
110             */
111            private String generate(int[] arr){
112                    Random dice = new Random();
113                    Set<Integer> s = new TreeSet<Integer>();  // use a set for no duplicates
114                    while(s.size() < STRESS_FACTOR){
115                            s.add(dice.nextInt(2000000)); // randomize the input
116                            
117                    }
118                    
119                    StringBuilder deathBed = new StringBuilder();
120                    int index = 0;
121                    Iterator<Integer> iter = s.iterator();
122                    while(iter.hasNext()){
123                            int next = iter.next();
124                            
125                            arr[index] = next;
126                            deathBed.append(next+",");        // record in string form
127                            index++;
128                    }
129                    // solve the fence post issue
130                    deathBed.deleteCharAt(deathBed.length()-1);
131                    return deathBed.toString();
132            }

184        /**
185         * helper to test false cases for contains
186         */
187        private void getNotContains(StreetNumberSet numSet, int[] arr){
188            List<Integer> in = new LinkedList<Integer>();
189            List<Integer> out = new LinkedList<Integer>();
190            for(int i = 0; i < arr.length; i++){
191                    in.add(arr[i]);
192            }
193            
194            int count = 0;
195            int index = 0;
196            while(count < 100){
197                    if(!in.contains(count)){
198                            out.add(count);
199                    }
200                    count++;
201            }
202            
203            for(Integer i : out){
204                    try{
205                            assertFalse("NumberSet should not contain number "+i+", which is outside " +
206                                            "of this NumberSet's range",
207                                    numSet.contains(i));
208                    }catch(Exception ex){
209                            fail(ex+" is an invalid response:  specification does not forbid " +
210                                            "any bad arguments for contains().");
211                    }
212            }
213            
214        }

255        /**
256         * helper to test statistics
257         */
258        private void getStats(StreetNumberSet numSet, int[] arr, int stat){
259            int count = 0;
260            int i;
261            for(int j = 0; j < arr.length; j++){
262                    i = arr[j];
263                    if(i < stat){
264                            count++;
265                    }
266            }
267            
268            try{
269                    assertEquals("statistics count should be "+count+", instead was "+
270                                    numSet.orderStatistic(stat), count, numSet.orderStatistic(stat));
271            }catch(Exception ex){
272                    fail(ex+" is an invalid response:  specification does not forbid " +
273                                    "arguments outside of address range for orderStatistics().");
274            }
275        }
