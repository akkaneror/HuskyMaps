001    package ps4.test;
002    
003    import ps4.*;
004    import junit.framework.*;
005    
006    public class StreetNumberSetImpTest extends TestCase {
007    
008            /** number sets */
009            private final StreetNumberSet numSetEq1;
010            private final StreetNumberSet numSetEq2;
011            private final StreetNumberSet numSetDiff1;
012            private final StreetNumberSet numSetDiff2;
013            private final StreetNumberSet numSetZero;
014            private final StreetNumberSet numSetBig;
015            
016            /** arrays of all numbers that should be present*/
017            private final int[] arrEq1 = {14,16,18,20,22};
018            private final int[] arrEq2 = {14,16,18,20,22};
019            private final int[] arrDiff1 = {14,15,16,17,18,19,20,22};
020            private final int[] arrDiff2 = {14,18,20,22};
021            private final int[] arrZero = {};
022            private final int[] arrBig = {15,8,10,12,7,9,11,0,13,18,5,42,44,57};
023            
024            
025            public StreetNumberSetImpTest(String name){
026                    super(name);
027                    
028                    // initialize ranges
029            String numEq1 = "14-22";
030            String numEq2 = "14-16,18,20-22";       
031            String numDiff1 = "14-18,15-19,20-22";
032            String numDiff2 = "14,18,20-22";
033            String numZero = "";
034            // non-ascending order seed string.  No specification about state of
035            // seed string on order of numbers!
036            String numBig = "15,8-12,7-13,0,18,5,42-44,57";   
037            
038            // initialize StreetNumberSets
039            numSetEq1 = new StreetNumberSet(numEq1);
040            numSetEq2 = new StreetNumberSet(numEq2);
041            numSetDiff1 = new StreetNumberSet(numDiff1);
042            numSetDiff2 = new StreetNumberSet(numDiff2);
043            numSetZero = new StreetNumberSet(numZero);
044            numSetBig = new StreetNumberSet(numBig);
045            
046            }
047            
048            
049            
050        
051        /**
052         * test size operation
053         */
054        public void testSize(){
055            
056            assertEquals("size should be "+getSize(arrEq1)+", instead was "+
057                            numSetEq1.size(), getSize(arrEq1), numSetEq1.size());
058            assertEquals("size should be "+getSize(arrDiff1)+", instead was "+
059                            numSetDiff1.size(), getSize(arrDiff1), numSetDiff1.size());
060            assertEquals("size should be "+getSize(arrBig)+", instead was "+
061                            numSetBig.size(), getSize(arrBig), numSetBig.size());
062            assertEquals("size should be "+getSize(arrZero)+", instead was "+
063                            numSetZero.size(), getSize(arrZero), numSetZero.size());
064        }
065        
066        
067        /**
068         * helper to test size
069         */
070            private int getSize(int[] arr){
071                int count = 0;
072                    for(int j = 0; j < arr.length; j++){
073                    count++;
074                }
075                    return count;
076            }
077        
078            
079        /**
080         * test hashCode
081         */
082        public void testHashCode(){
083            assertTrue("numbersets covering the same number range, even with different " +
084                            "string seeds, should have the same hash code", numSetEq1.hashCode() == 
085                            numSetEq2.hashCode() );
086        }
087            
088    
089            
090    }
