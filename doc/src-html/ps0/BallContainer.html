004    package ps0;

006    import java.lang.Iterable;
007    import java.util.Set;
008    import java.util.LinkedHashSet;
009    import java.util.Iterator;
010    import java.util.Collections;
011    
012    /**
013     * This is a container can be used to contain Balls.
014     * A given Ball may only appear in a BallContainer once.
015     */
016    public class BallContainer implements Iterable<Ball> {
017    
018        // Contents of the box.
019        private Set<Ball> contents;
020        // Total number of balls in collection
021        private int size;
022    
023        /**
024         * Constructor that creates a new ballcontainer.
025         */
026        public BallContainer() {
027            contents = new LinkedHashSet<Ball>();
028            size = 0;
029        }
030    
031        /**
032         * Implements the Iterable interface for this container.
033         * @return an Iterator over the Ball objects contained
034         * in this container.
035         */
036        public Iterator<Ball> iterator() {
037            // If we just returned the iterator of "contents", a client
038            // could call the remove() method on the iterator and modify
039            // it behind our backs.  Instead, we wrap contents in an
040            // "unmodifiable set"; calling remove() on this iterator
041            // throws an exception.  This is an example of avoiding
042            // "representation exposure."  You will learn more about this
043            // concept later in the course.
044            return Collections.unmodifiableSet(contents).iterator();
045        }
046    
047        /**
048         * Adds a ball to the container. This method returns <tt>true</tt>
049         * if ball was successfully added to the container, i.e. ball is
050         * not already in the container. Of course, you are allowed to put
051         * a Ball into a container only once. Hence, this method returns
052         * <tt>false</tt>, if ball is already in the container.
053         * @param b Ball to be added.
054         * @return true if ball was successfully added to the container,
055         * i.e. ball is not already in the container. Returns false, if ball is
056         * already in the container.
057         */
058        public boolean add(Ball b) {
059            if(!contents.contains(b)){ // check if ball *b is in container already
060                    contents.add(b); // if not, add ball *b and return true
061                    size++;
062                    return true;
063            }
064            return false;
065        }
066    
067        /**
068         * Removes a ball from the container. This method returns
069         * <tt>true</tt> if ball was successfully removed from the
070         * container, i.e. ball is actually in the container. You cannot
071         * remove a Ball if it is not already in the container and so ths
072         * method will return <tt>false</tt>, otherwise.
073         * @param b Ball to be removed.
074         * @return true if ball was successfully removed from the container,
075         * i.e. ball is actually in the container. Returns false, if ball is not
076         * in the container.
077         */
078        public boolean remove(Ball b) {
079            if(contents.contains(b)){ //check if ball *b is present
080                    contents.remove(b);
081                    size--;
082                    return true;
083            }
084            return false;
085        }
086    
087        /**
088         * Each Ball has a volume. This method returns the total volume of
089         * all the Balls in the container.
090         * @return the volume of the contents of the container.
091         */
092        public double getVolume() {
093            double sum = 0;
094            Iterator<Ball> iter = iterator();
095            
096            while(iter.hasNext()){ //iterate through *contents using an iterator
097                    sum += iter.next().getVolume(); //accumulate a running sum
098            }
099            return sum;
100        }
101    
102        /**
103         * Returns the number of Balls in this container.
104         * @return the number of Balls in this container.
105         */
106        public int size() {
107            return size;
108        }
109    
110        /**
111         * Empties the container, i.e. removes all its contents.
112         */
113        public void clear() {
114            contents.removeAll(contents);
115            size = 0;
116        }
117    
118        /**
119         * This method returns <tt>true</tt> if this container contains
120         * the specified Ball. It will return <tt>false</tt> otherwise.
121         * @param b Ball to be checked if its in container
122         * @return true if this container contains the specified Ball. Returns
123         * false, otherwise.
124         */
125        public boolean contains(Ball b) {
126            return contents.contains(b);
127        }
128    
129    }
