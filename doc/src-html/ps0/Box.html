006    import java.lang.Iterable;
007    import java.util.Iterator;
008    import java.util.*;
009    
010    /**
011     * This is a container can be used to contain Balls. The key
012     * difference between a BallContainer and a Box is that a Box has a
013     * finite volume. Once a box is full, we cannot put in more Balls.
014     */
015    public class Box implements Iterable<Ball> {
016    
017        /**
018         * ballContainer is used to internally store balls for this Box
019         */
020        private BallContainer ballContainer = new BallContainer();
021        /**
022         * capacity of the Box
023         */
024        private double maxVolume;
025    
026        /**
027         * Constructor that creates a new box.
028         * @param maxVolume Total volume of balls that this box can contain.
029         */
030        public Box(double maxVolume) {
031            this.maxVolume = maxVolume;
032        }
033    
034        /**
035         * Implements the Iterable interface for this box.
036         * @return an Iterator over the Ball objects contained
037         * in this box.
038         */
039        public Iterator<Ball> iterator() {
040            return ballContainer.iterator();
041        }
042    
043    
044        /**
045         * This method is used to add Ball objects to this box of
046         * finite volume.  The method returns true if a ball is
047         * successfully added to the box, i.e., ball is not already in the
048         * box and if the box is not already full; and it returns false,
049         * if ball is already in the box or if the box is too full to
050         * contain the new ball.
051         * @param b Ball to be added.
052         * @return true if ball was successfully added to the box,
053         * i.e. ball is not already in the box and if the box is not
054         * already full. Returns false, if ball is already in the box or
055         * if the box is too full to contain the new ball.
056         */
057        public boolean add(Ball b) {
058            if(ballContainer.getVolume() + b.getVolume() <= maxVolume){ // if ball can fit
059                    return ballContainer.add(b); // refer to ballContainer's search for duplicate balls
060            }
061            return false;
062        }
063    
064        /**
065         * This method returns an iterator that returns all the balls in
066         * this box in ascending size, i.e., return the smallest Ball
067         * first, followed by Balls of increasing size.
068         * @return an iterator that returns all the balls in this box in
069         * ascending size.
070         */
071        public Iterator<Ball> getBallsFromSmallest() {
072            Set<Ball> unordered = new LinkedHashSet<Ball>();
073            
074            for(Ball b: ballContainer){
075                    unordered.add(b);
076            }
077            
078            Set<Ball> ordered = new LinkedHashSet<Ball>();//make new set        
079            Ball smallest; //make ball pointer
080                    
081            while(unordered.size() != 0){ //while ballContainer is not empty                
082                    Iterator<Ball> iter = unordered.iterator();//make ballContainer iterator
083                    smallest = iter.next(); //set ball pointer to first element
084                    Ball curr;
085                    
086                    while(iter.hasNext()){ //iterate over unordered
087                            curr = iter.next();
088                            if(curr.getVolume() < smallest.getVolume()){
089                                    smallest = curr; //set ball pointer to smaller of two balls
090                            }
091                    }
092                    
093                    unordered.remove(smallest); //remove from old set, add to new set
094                    ordered.add(smallest);
095            }
096                    
097            return ordered.iterator();
098        }
099    
100        /**
101         * Removes a ball from the box. This method returns
102         * <tt>true</tt> if ball was successfully removed from the
103         * container, i.e. ball is actually in the box. You cannot
104         * remove a Ball if it is not already in the box and so ths
105         * method will return <tt>false</tt>, otherwise.
106         * @param b Ball to be removed.
107         * @return true if ball was successfully removed from the box,
108         * i.e. ball is actually in the box. Returns false, if ball is not
109         * in the box.
110         */
111        public boolean remove(Ball b) {
112            return ballContainer.remove(b);
113        }
114    
115        /**
116         * Each Ball has a volume. This method returns the total volume of
117         * all the Balls in the box.
118         * @return the volume of the contents of the box.
119         */
120        public double getVolume() {
121           return ballContainer.getVolume();
122        }
123    
124        /**
125         * Returns the number of Balls in this box.
126         * @return the number of Balls in this box.
127         */
128        public int size() {
129            return ballContainer.size();
130        }
131    
132        /**
133         * Empties the box, i.e. removes all its contents.
134         */
135        public void clear() {
136            ballContainer.clear();
137        }
138    
139        /**
140         * This method returns <tt>true</tt> if this box contains
141         * the specified Ball. It will return <tt>false</tt> otherwise.
142         * @param b Ball to be checked if its in box
143         * @return true if this box contains the specified Ball. Returns
144         * false, otherwise.
145         */
146        public boolean contains(Ball b) {
147            return ballContainer.contains(b);
148        }
149    
150    }
