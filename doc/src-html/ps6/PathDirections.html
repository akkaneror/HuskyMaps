001    package ps6;
002    
003    import java.util.Collections;
004    import java.util.Iterator;
005    import java.util.List;
006    
007    import ps2.Route;
008    
009    /**
010     *   A PathDirections is a description of how to travel from one Address to
011     *   another.
012     *   @specfield start      : Address // starting point
013     *   @specfield end        : Address // destination
014     *   @specfield length     : double  // traveling distance, in miles (not rounded)
015     *   @specfield route      : Route  // this sequence of segments the directions
016     *                                                              // travel over
017     *   @specfield directions : sequence[String] //
018     *   directions for traveling from start to end.  Each
019     *   String in the sequence must be one line of directions, in the format
020     *   specified in <a href="../../psets/ps6/ps6.html#dirs-format">PS6
021     *   text based interface</a>, <b>excluding</b> the first line which
022     *   states where you start <b>and</b> the final line which gives the length/time
023     *   of the trip.  The strings <b>must not</b> include newlines at their
024     *   ends.  The order of the Strings in the sequence must correspond to the order
025     *   of the lines in the directions from start to end.
026     */
027    
028    public class PathDirections implements Directions{
029    
030            private final Address start;
031            private final Address end;
032            private final Route route;
033            private final List<String> directions;
034            
035            /**
036             * @requires directions != null && route != null && start != null && end != null
037             */
038            public PathDirections(List<String> directions, Route route, Address start, Address end){
039                    if(directions == null || route == null || start == null || end == null){
040                            throw new IllegalArgumentException("Specified arguments must not be null");
041                    }
042                    
043                    this.start = start;
044                    this.end = end;
045                    this.route = route;
046                    this.directions = Collections.unmodifiableList(directions);
047            }
048            
049            
050        /** @return this.length. */
051        public double getLength (){
052            return route.getLength();
053        }
054    
055        /** @return an Iterator of Strings over this.directions */
056        public Iterator<String> iterator(){
057            return directions.iterator();
058        }
059    
060        /** @return this.start */
061        public Address getStart (){
062            return start;
063        }
064    
065        /** @return this.end */
066        public Address getEnd (){
067            return end;
068        }
069        
070        /** @return this.route */
071        public Route getRoute(){
072            return route;
073        }
074        
075        public String toString(){
076            String result = "Start at "+start+"\n";
077                    
078                    Iterator<String> iter = iterator();
079                    while(iter.hasNext()){
080                            result += iter.next()+"\n";
081                    }
082                    
083                    return result + "Trip length: "+route.getLength();
084        }
085            
086    }
