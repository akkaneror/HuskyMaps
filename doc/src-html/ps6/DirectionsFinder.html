018    /**
019     *   A DirectionsFinder produces directions for traveling from one
020     *   address to another in a particular geographical area.  Geographical information
021     *   is obtained from a valid Tiger Database Directory, a directory of geographical 
022     *   street and address information.
023     *   @specfield segments : set[StreetSegment]  // the set of StreetSegments
024     *   which represents the geographical area for which this DirectionsFinder
025     *   produces directions
026     *   @specfield geographicMap : graph[StreetSegment]  // the graph representing the
027     *   geographic area to be operated over by DirectionsFinder.
028     *   @specfield streetIndex : Map[String, Set[StreetSegment] ]  // an index of all 
029     *   street names within this geographic area, with each name mapped to all street 
030     *   segments with that name.
031     *   @specfield zipcodeFilter : Collection[String]  // collection of zipcodes that 
032     *   this geographic area is constrained to, or if nothing contained, then there 
033     *   is no zipcode limit to the area's coverage.
034     *   
035     */
036    public class DirectionsFinder
037    {
038            /*
039             * Representation Invariant
040             *      geographicMap:
041             *              must not contain duplicate StreetSegments
042             *      streetIndex:
043             *              for any Set<StreetSegment> s in streetIndex.values():
044             *                      s must not contain a StreetSegment with name "(unnamed street)"
045             *                      &&  for at least one zipcode z of zipcodeSet, s.checkZip(z) == true 
046             * 
047             * 
048             * Abstraction Function
049             *  set[StreetSegment] segments:
050             *      Collection geographicMap.values() contains all StreetSegments this DirectionsFinder
051             *      produces directions over
052             *      Streets:
053             *              each element of geographicMap.nodeSet() is a unique street in this geographic
054             *              area.
055             *      Intersections:
056             *              let Set<StreetSegment> s = geographicMap.edgeSet(StreetSegment p)
057             *              for all segments p contained in the graph
058             *                      then street p shares an intersection with each street c of s 
059             *                      &&  it is a valid driving/walking maneuver to traverse from street p
060             *                      to street c at that intersection.
061             * 
062             */
063            
064            /** switch to turn on status reporting for the class during runtime, printed to the console */
065            public static boolean VERBOSE = false;
066            /** switch to turn on debugging reporting for the class during runtime, printed to the console */
067            private static boolean DEBUG = false;
068            /** switch to turn on additional debugging reporting for the class during runtime, printed to the console */
069            private static boolean DEBUG_FRACTIONAL = false;
070            
071            /** constant to aid in error-check-identification of a valid address */
072            public static final int ADDRESS_VALID = 0;
073            /** constant to aid in error-check-identification of an invalid zipcode */
074            public static final int ADDRESS_ZIPCODE_INVALID = -1;
075            /** constant to aid in error-check-identification of an invalid street */
076            public static final int ADDRESS_STREET_INVALID = -2;
077            /** constant to aid in error-check-identification of an invalid address number */
078            public static final int ADDRESS_NUM_INVALID = -3;
079            /** constant to aid in error-check-identification of a null address */
080            public static final int ADDRESS_NULL = -4;

085             * of this geographic area */
086            private Graph<StreetSegment> geographicMap;
087            /** Index of street names to all StreetSegments with that name */
088            private Map<String, Set<StreetSegment> > streetIndex;
089            /** Map of GeoSegments to corresponding StreetSegments, to avoid casting, and
090             * all contained within the graph */
091            private Map<GeoSegment, StreetSegment> segments;
092            /** Map of fractional StreetSegments to corresponding full StreetSegments that each
093             * was fractioned over, where full segments are present on the graph. */
094            private Map<StreetSegment, StreetSegment> fractionalEntryIndex;
095            /** Set of zipcodes that are contained in this geographic area. */
096            private Set<String> zipcodeSet;

103        /** Produces a DirectionsFinder for a given database.
104         * @param databaseName the database to find directions in
105         * @param zf a collection of zipcodes used for zipcode filtering.
106         * Only matching zipcodes will be used.
107         * If zf is empty or null then there is no zipcode filtering for
108         * the database.
109         * @return A DirectionsFinder whose set of segments is
110         *  { s | s in (new StreetSegReader(databaseName)).streetSegs () &&
111         *  (s.leftZip in zf || s.rightZip in zf || (s.leftZip.trim () ==
112         *  s.rightZip.trim () == "") || zf == null) }
113         * @throws InvalidDatabaseException if databaseName does not refer to a
114         *  valid tiger database.
115         */
116        public static DirectionsFinder getDirectionsFinder(String databaseName, Collection<String> zf)
117                    throws InvalidDatabaseException {
118            
119            StreetSegReader source;
120            try{
121                    source = new StreetSegReader(databaseName);
122                            //throws InvalidSourceException if specified source database is invalid
123            }catch(Exception ex){
124                    throw new InvalidDatabaseException("Specified Tiger Database Directory invalid");
125            }
126                
127            // <<<< create a street index >>>>
128            // indexing structure of street names to all the street segments with that name
129            Map<String, Set<StreetSegment> > streetIndex = new HashMap<String, Set<StreetSegment> >();
130            Map<GeoSegment, StreetSegment> segments = new HashMap<GeoSegment, StreetSegment>();
131            Set<String> zipcodeSet = new HashSet<String>();
132            if(zf != null && !zf.isEmpty()){
133                    zipcodeSet.addAll(zf);
134            }
135            
136            if(VERBOSE){ System.out.println("--> Filtering zipcodes of "+zf+" ..."); }
137            
138            long startNodes = System.currentTimeMillis();
139            // for all segments of the database
140            Iterator<StreetSegment> iter = source.streetSegments();
141            while(iter.hasNext()){
142                    StreetSegment stS = iter.next();
143                    
144                    // if within the specified zipcode boundaries 
145                    boolean rightZipIn = withinZip(stS, stS.getRightZip(), zf);
146                    boolean leftZipIn = withinZip(stS, stS.getLeftZip(), zf);
147                    if(rightZipIn || leftZipIn){
148                            
149                            // add to the total collection
150                            segments.put(stS, stS);
151                            StreetSegment rev = stS.reverse();
152                            segments.put(rev, rev);
153                            if(zf == null || zf.isEmpty()){
154                                    zipcodeSet.add(stS.getLeftZip());
155                                    zipcodeSet.add(stS.getRightZip()); 
156                            }
157                            
158                            // for the segment index
159                                    String name = stS.getName();
160                                    
161                                    // don't include segments of "(unnamed street)" in the index, so they can't be found
162                                    if(!name.equals("(unnamed street)")){
163                                            
164                                            // expand the mapping
165                                    Set<StreetSegment> t = !streetIndex.containsKey(name) ? 
166                                                    new HashSet<StreetSegment>() : streetIndex.get(name);
167                                    t.add(stS);
168                                    streetIndex.put(name, t);
169                                    }
170                    }
171                    }
172            long endNodes = System.currentTimeMillis();
173            if(VERBOSE) System.out.println("Time for populating: " + (endNodes-startNodes)/1000.0);
174            
175            DirectionsFinder dirFind = new DirectionsFinder();
176            
177            if(VERBOSE){ System.out.println("--> Constructing the Graph ..."); }
178            
179            // use a factory class to produce a graph of the geographic area
180            dirFind.geographicMap = GraphGenerator.produceGraph(segments.values());
181            
182            dirFind.streetIndex = Collections.unmodifiableMap(streetIndex);
183                    dirFind.segments = Collections.unmodifiableMap(segments);
184                    dirFind.fractionalEntryIndex = new HashMap<StreetSegment, StreetSegment>();
185                    dirFind.zipcodeSet = Collections.unmodifiableSet(zipcodeSet);
186    
187                    return dirFind;
188        }

191        /** helper to check segments against the zip code filter*/
192        private static boolean withinZip(StreetSegment gs, String zip, Collection<String> zf){
193            return  (zf == null || zf.isEmpty() || 
194                            zf.contains(zip) || 
195                            (gs.getRightZip().equals("") && gs.getLeftZip().equals("")) );
196        }

199        /** Gets directions from start to end.
200         * @requires routeFormat != null
201         * @throws InvalidAddressException if start or end is not in the
202         *  geographical area covered by this DirectionsFinder.
203         * @throws NoPathException if no path could be found from start to end.
204         * @throws IllegalArgumentException if the specified RouteFormatter is null
205         * @return A Directions providing directions of travel from start to end.
206         */
207        public Directions getDirections(Address start, Address end, ps2.RouteFormatter routeFormat)
208                    throws InvalidAddressException, NoPathException {
209            
210            if(routeFormat == null){
211                    throw new IllegalArgumentException("RouteFormatter must not be null");
212            }else{
213                    int first = checkAddress(start);
214                    int second = checkAddress(end);
215                    if(first != ADDRESS_VALID || second != ADDRESS_VALID){
216                            throw new InvalidAddressException("Address does not exist (should have caught this in TextUI): "+
217                                                    "start = "+first+", end = "+second);
218                    }
219            }
220            
221            if(VERBOSE){ System.out.println("--> Calculating the shortest path ..."); }
222            
223            StreetEntry startEntry = new StreetEntry(findAddress(start), start, false); 
224            StreetEntry endEntry = new StreetEntry(findAddress(end), end, true);
225            
226            
227            // calculate the shortest path
228                    Path<StreetSegment, StreetPath> path = calculateShortestPath(startEntry, endEntry);
229                    
230            // Check if no path was found
231            if(path == null){
232                    throw new NoPathException("No path found");
233            }
234            
235            if(VERBOSE){ System.out.println("--> Processing the found path ..."); }
236            
237            // make new Route using path.iterator(), catalog first and last streets
238            Iterator<StreetSegment> iter = path.iterator();
239            StreetSegment next = iter.next();
240            startEntry = new StreetEntry(next, start, false);
241            Route route = new Route(next);
242            while(iter.hasNext()){
243                    next = iter.next();
244                    route = route.addSegment(next);
245            }
246            endEntry = new StreetEntry(next, end, true);
247            
248            if(DEBUG){
249                    System.out.println("found route contains the following:");
250                    for(GeoSegment gs : route.getGeoSegments()){
251                            System.out.println(gs);
252                    }
253                    System.out.println("route done");
254            }
255            
256            // >>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>> CONSISTENCY CHECK FOR PATHFINDER ON SAME-STREET ADDRESSES
257            if(startEntry.getStreet().equals(endEntry.getStreet().reverse())){
258                    throw new RuntimeException("INCONSISTENCY for same-street addresses PathFinder " +
259                                    "should return single-street path");
260            }
261            
262            
263            List<String> directions = processDirections(route, startEntry, endEntry, routeFormat); //new LinkedList<String>();
264    
265            return new PathDirections(directions, route, start, end);
266            
267        }

269        /** returns true if the specified address addr is a valid address, that is
270         * streetIndex contains a StreetSegment sts such that 
271         *          sts.name == addr.name &&
272         *          sts.contains(addr.number,  addr.zipcode)
273         * @return an int value equal to class constants defined for this class:
274         *          ADDRESS_VALID if the specified address is a valid address
275         *          ADDRESS_ZIPCODE_INVALID if the address zipcode is invalid
276         *          ADDRESS_STREET_INVALID if the address street name is not within the address zipcode
277         *          ADDRESS_NUM_INVALID if the address number is invalid for an existing street
278         *          ADDRESS_NULL if the address object is null
279         * */
280        public int checkAddress(Address addr){
281            if(addr == null){
282                    return ADDRESS_NULL;
283            }
284            
285            // if this address's zip passes the zipcode filter
286            String addrZip = addr.getZipcode();
287            if(!zipcodeSet.contains(addrZip)){
288                    return ADDRESS_ZIPCODE_INVALID;
289            }
290            
291            // if this address's street name is a real street
292            String addrName = addr.getName();
293            if(!streetIndex.containsKey(addrName)){
294                    return ADDRESS_STREET_INVALID;
295            }
296            
297            boolean zipHasStreet = false;
298            int addrNum = addr.getNum();
299            // for all streets of this name
300            for(StreetSegment stS : streetIndex.get(addrName)){
301                    
302                    // if this street exists in the specified zipcode
303                    if(stS.checkZip(addrZip)){
304                            zipHasStreet = true;
305                    
306                            // if this address is found on a street with the right zipcode
307                            if(stS.contains(addrNum, addrZip)){
308                                    return ADDRESS_VALID;
309                            }
310                    }
311            }
312            // if the street was in the zipcode, then only the address was wrong 
313            return zipHasStreet ? ADDRESS_NUM_INVALID : ADDRESS_STREET_INVALID;
314        }

319        /**
320         * helper to search the StreetSegment index
321         * @requires specified address must be valid
322         * @return the StreetSegment that the specified address is located on.
323         * @throws IllegalArgumentException if address not found
324         */
325        private StreetSegment findAddress(Address addr){
326            for(StreetSegment stS : streetIndex.get(addr.getName())){
327                    if(stS.contains(addr.getNum(), addr.getZipcode())){
328                            return stS;
329                    }
330            }
331            throw new IllegalArgumentException("Address-not-found inconsistancy");
332        }

334        /**
335         * Searches for the StreetSegment companion to the specified GeoSegment using
336         * the segment indexes built within this instance of DirectionsFinder.  Designed
337         * to find full-length StreetSegments for fractional segments as well (as fractional
338         * segments are never contained within the underlying Graph collection) . 
339         * @requires specified GeoSegment must correspond to a street within the graph
340         * whether a whole or a fractional segment
341         * @return the StreetSegment that the specified GeoSegment refers to, even if that
342         * GeoSegment is fractional
343         * @throws IllegalArgumentException if street not found
344         */
345        public StreetSegment findFullSegment(GeoSegment gs){
346            StreetSegment street = segments.get(gs);
347            // fractional case
348                    street = (street == null ? fractionalEntryIndex.get(gs) : street );
349                    // runtime error case
350                    if(street == null) throw new IllegalArgumentException("Passed in segment "+gs+
351                                    " which was not an element of this instance of DirectionsFinder"); 
352            return street;
353        }

356        /** Sets up parameters for StreetPath and calls PathFinder to calculate 
357         * the travel path */
358        private Path<StreetSegment, StreetPath> calculateShortestPath(StreetEntry depart,
359                    StreetEntry destiny){
360            
361            StreetEntry departRev = depart.reverse();
362            StreetEntry destinyRev = destiny.reverse();
363            catalogEntrys(depart, destiny);
364            catalogEntrys(departRev, destinyRev);
365            if(DEBUG_FRACTIONAL){
366                    for(StreetSegment stS : fractionalEntryIndex.keySet()){
367                            System.out.println("Recorded fractional segment "+stS+" on street "+
368                                            fractionalEntryIndex.get(stS));
369                    }
370            }       
371            
372            // goals = make a set of 2 (norm + rev) end segments based on end address
373            Set<StreetEntry> goals = new HashSet<StreetEntry>();
374            goals.add(destiny);
375            goals.add(destinyRev);
376            
377            // starts = make a set of 2 (norm + rev) RoutePaths based on start address
378            Set<Path<StreetSegment, StreetPath> > starts = new HashSet<Path<StreetSegment, StreetPath>>();
379            starts.add(new StreetPath(depart, goals, geographicMap));
380            starts.add(new StreetPath(departRev, goals, geographicMap));
381            
382            Set<StreetSegment> streetGoals = new HashSet<StreetSegment>();
383            for(StreetEntry se : goals){
384                    streetGoals.add(se.getStreet());
385            }
386            
387            return PathFinder.findPath(geographicMap, starts, streetGoals);
388        }

391        /** helper to catalog fractional StreetSegments generated by StreetEntry objects */
392        private void catalogEntrys(StreetEntry start, StreetEntry end){
393            StreetSegment first = start.getStreet();
394            StreetSegment second = end.getStreet();
395            if(first.equals(second) || first.equals(second.reverse())){
396                    // catalog same street case
397                    fractionalEntryIndex.put(start.getSameStreetFractional(end), start.getStreet());
398            }else{
399                    // catalog the fractional segments of the start address 
400                    fractionalEntryIndex.put(start.getFractionalSegment(), start.getStreet());
401                    fractionalEntryIndex.put(start.getFractionalSegment().reverse(), start.getStreet().reverse());
402                    
403                    // catalog the fractional segments of the end address
404                    fractionalEntryIndex.put(end.getFractionalSegment(), end.getStreet());
405                    fractionalEntryIndex.put(end.getFractionalSegment().reverse(), end.getStreet().reverse());
406            }
407    
408        }

414            /** helper to determine the start heading of travel when beginning from an 
415             * address on a street */
416        private double addressHeading(StreetSegment street, Address addr){
417            double heading = street.getHeading();
418            // rotate based on the street side the address is on
419            heading += (street.rightSideContains(addr.getNum(), addr.getZipcode())) ? -90 : 90;
420            heading += heading > 360 ? -360 : 0;  // make sure rep invariant is held
421            heading += heading < 0 ? 360 : 0;
422            return heading;
423        }

426        /** Generates directions for the same-segment directions case. Directions are hard-coded,
427         * and do not rely on a RouteFormatter */
428        private String processSameStreet(StreetSegment street, Address start, Address end, 
429                    ps2.RouteFormatter format){
430    
431            // if start address farther down the street
432            boolean endBeforeStart = (((getFractDistance(street, start) > getFractDistance(street, end))));
433    
434            
435            // figure out start turn
436            String startTurn;
437            if((street.rightSideContains(start.getNum(), start.getZipcode()) && endBeforeStart) ||
438                            (!street.rightSideContains(start.getNum(), start.getZipcode()) && !endBeforeStart)){
439                    startTurn = "Turn left ";
440            }else{
441                    startTurn = "Turn right ";
442            }
443    
444            // figure out end turn
445            String endTurn;
446            if((street.rightSideContains(end.getNum(), end.getZipcode()) && endBeforeStart) ||
447                            (!street.rightSideContains(end.getNum(), end.getZipcode()) && !endBeforeStart)){
448                    endTurn = "left";
449            }else{
450                    endTurn = "right";
451            }
452            
453            double length = street.getLength();
454            String travel = (format instanceof WalkingRouteFormatter ? "walk for "+Math.round(length*20) +" minutes." : 
455                            "go "+Math.round(length*10)/10.0+" miles." );
456                    
457                    return startTurn+"onto "+street.getName()+" and "+travel+"\n"+end+" is on your " + endTurn;
458            
459        }

461        /** Generates a list of directions from a complete route, ready to be passed to a 
462         * Directions container object */
463        private List<String> processDirections(Route route, StreetEntry startEntry, StreetEntry endEntry, 
464                    ps2.RouteFormatter routeFormat){
465            
466            // catch the same-segment case
467            StreetSegment sameStreet = null;
468            if(startEntry.getStreet().equals(endEntry.getStreet())) 
469            {
470                    // get the fractional street with start address before end address
471                    sameStreet = startEntry.getSameStreetFractional(endEntry);
472                    route = new Route(sameStreet);
473            }
474            
475            // get directions over the route
476                String directLine = routeFormat.computeDirections(route, addressHeading( (sameStreet != null ? 
477                            sameStreet : startEntry.getStreet() ), startEntry.getAddress()));
478                
479                // add the last arrival line, taking into account same-segment possibility
480                Address endAddress = endEntry.getAddress();
481            StreetSegment endStreet = (sameStreet != null ? sameStreet : endEntry.getStreet() );
482                directLine += endAddress+" is on your " +
483                            (endStreet.rightSideContains(endAddress.getNum(), endAddress.getZipcode() ) ? 
484                                            "right" : "left");
485                
486                    List<String> directions = new LinkedList<String>();
487                    String[] lines = directLine.split("\n");
488                    
489                    // add the  directions
490                    // for every line, that is an element
491                    for(int i = 0; i < lines.length; i++){
492                            directions.add(lines[i]);
493                    }
494            
495                    return directions;
496        }

499        /** helper to correctly determine fractional distance, even in case of
500         *  same address number on both sides */
501        private double getFractDistance(StreetSegment stS, Address addr){
502            int num = addr.getNum();
503            // because fractionDist is only guaranteed to grab the right-side distance 
504            // in case of same number on both sides
505            if(stS.rightSideContains(num, addr.getZipcode())){
506                    return stS.fractionDist(num);
507            }else{  // reverse to treat segment's left-side as a fractionDist right-side  
508                    return 1.0 - stS.reverse().fractionDist(num);
509            }
510        }

515        /**
516         * Determines nearby segments of the Graph to the specified Route.
517         * @returns collection of all segments of the Graph that fall within the specified 
518         * latitude and longitude bounds and connect somehow to the specified Route
519         */
520        public Collection<StreetSegment> getContext(int[] minMaxLatLon, Route rt){
521            if(DEBUG || DEBUG_FRACTIONAL){
522                    System.out.println("found route contains the following:");
523                    for(GeoSegment gs : rt.getGeoSegments()){
524                            System.out.println(gs);
525                    }
526                    System.out.println("route done");
527            }
528            
529            Set<StreetSegment> map = new HashSet<StreetSegment>(); 
530            
531            Stack<StreetSegment> explore = new Stack<StreetSegment>();
532            explore.push( findFullSegment(rt.getEndingGeoSegment()) );
533            
534                    if(DEBUG) System.out.println("explore segment "+rt.getEndingGeoSegment());
535            
536            for(GeoSegment gs : rt.getGeoSegments()){
537                    explore.push( findFullSegment(gs) );
538                    explore.push( findFullSegment(gs.reverse()) );
539                            if(DEBUG) System.out.println("explore segment "+gs);
540            }
541            
542            // for all segments needed to explore
543            while(!explore.isEmpty()){
544                    StreetSegment gs = explore.pop();
545                    // if this is a unique segment and it starts within the bounds
546                    if(!map.contains(gs) && !map.contains(gs.reverse()) && withinContext(minMaxLatLon, gs.getP1())){
547                            map.add(gs);
548                            if(DEBUG) System.out.println("added to map segment "+gs+" with bounds of minMaxLatLon");
549                    }
550                    
551                    // even if it isn't, still need to examine this segment's P2 children
552                    if(withinContext(minMaxLatLon, gs.getP2())){   // if it's P2 is within the bounds
553                            for(StreetSegment edge : geographicMap.edgeSet(gs)){
554                                    if(!map.contains(edge) && !map.contains(edge.reverse())){
555                                            explore.push(edge);     // explore all children
556                                            if(DEBUG) System.out.println("explore segment "+edge);
557                                    }
558                            }
559                    }
560                    
561            }
562            
563            return map;
564        }

567        /** helper to determine if a GeoPoint is within the specified latitude and
568         * longitude bounds */
569        private boolean withinContext(int[] bounds, GeoPoint gp){
570            return gp.getLatitude() > bounds[0] && gp.getLatitude() < bounds[1] &&
571            gp.getLongitude() > bounds[2] && gp.getLongitude() < bounds[3];
572        }
