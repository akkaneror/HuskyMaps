001    package ps6;
002    import java.util.Iterator;
003    
004    import ps2.Route;
005    
006    /**
007     *   A Directions is a description of how to travel from one Address to
008     *   another.
009     *   @specfield start      : Address // starting point
010     *   @specfield end        : Address // destination
011     *   @specfield length     : double  // traveling distance, in miles (not rounded)
012     *   @specfield route      : Route  // this sequence of segments the directions
013     *                                                              // travel over
014     *   @specfield directions : sequence[String] //
015     *   directions for traveling from start to end.  Each
016     *   String in the sequence must be one line of directions, in the format
017     *   specified in <a href="../../psets/ps6/ps6.html#dirs-format">PS6
018     *   text based interface</a>, <b>excluding</b> the first line which
019     *   states where you start <b>and</b> the final line which gives the length/time
020     *   of the trip.  The strings <b>must not</b> include newlines at their
021     *   ends.  The order of the Strings in the sequence must correspond to the order
022     *   of the lines in the directions from start to end.
023     */
024    public interface Directions extends Iterable<String> {
025    
026        // a Directions is the result of a query to a DirectionsFinder
027    
028        /** @return this.length. */
029        public double getLength ();
030    
031        /** @return an Iterator of Strings over this.directions */
032        public Iterator<String> iterator();
033    
034        /** @return this.start */
035        public Address getStart ();
036    
037        /** @return this.end */
038        public Address getEnd ();
039        
040        /** @return this.route */
041        public Route getRoute ();
042    }
